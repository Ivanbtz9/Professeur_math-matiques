\begin{exer}\textit{Autour de la suite de Fibonacci}\\
	On définit la suite de Fibonacci $(F_n )_{n\in\N}$ par $F_0 = 0$, $F_1 = 1$ et la relation de récurrence : $$\forall n\in\N , \  F_{n+2} = F_{n+1} + F_n$$
	\begin{enumerate}
		\item Déterminer la liste des $10$ premiers nombres de Fibonacci (de $F_1$ à $F_{10}$ ).
		\item Montrer que : $\forall n\in \N , n\geq 6 \alors  F_n > n$.\\
		Que peut-on en déduire pour la suite $(F_n )_{n\in\N}$?
		\item Montrer que : $\forall n\in \N, \ n\geq 2 \alors F_n < F_{n+1} \leq 2F_n$. 
		\item Montrer que : $\forall n\in \N^*,\sum\limits_{k=1}^{n}F_k = F_{n+2}-1$
		\item Montrer que : $\forall n\in \N^*,\sum\limits_{k=1}^{n}F_k^2 = F_nF_{n+1}$
		\item Montrer que : $\forall n\in \N^*,\sum\limits_{k=1}^{2n-1}F_kF_{k+1} =F_{2n}^2$
		\item Montrer que : $\forall n\in \N^*,F_{2n} = F_{n+1}^2-F_{n-1}^2$ et $F_{2n+1}= F_{n+1}^2+ F_n^2$.
		\item Montrer que : $\forall n\in \N^*,\sum\limits_{k=0}^{n} \begin{pmatrix} 2n-k\\k\end{pmatrix} = F_{2n+1}$ et $\sum\limits_{k=0}^{n} \begin{pmatrix} 2n+1-k\\k\end{pmatrix} = F_{2n+2}$
		\item Montrer que si l'on pose: $\varphi = \dfrac{1+\sqrt{5}}{2}$ : $\forall n\in \N^*, F_{n-1}+ \varphi F_n = \varphi^n.$
		\item  Montrer que si l'on pose: $\overline{\varphi} = \dfrac{1-\sqrt{5}}{2}$ : $\forall n\in \N, F_n = \dfrac{\varphi^n - \overline{\varphi}^n}{\varphi- \overline{\varphi}}.$
	\end{enumerate}
\end{exer}

\begin{exer}\textit{$\mathbb{N}$ et $\mathbb{N}^{2}$ sont en bijection}
\begin{enumerate}


\item  Construire une bijection entre $\mathbb{N}$ et $\mathbb{N}^{*}$
\item  Construire une bijection entre $\mathbb{N}$ et $3 \mathbb{N} = \{ n \in \mathbb{N}, 3 \text{ divise } n \}$ 
\item Construire une bijection entre $\mathbb{N}$ et $\mathbb{Z}$
\item  Le but de cette question est de montrer qu'il existe une bijection entre $\mathbb{N}$ et $\mathbb{N}^2$. Soit $f : \mathbb{N} \longrightarrow \mathbb{N}^2$ l'application définie de la manière suivante :
    $$ \forall k \in \mathbb{N}, \forall n \in [[ \dfrac{k(k+1)}{2}, \dfrac{(k+1)(k+2)}{2} - 1 |], f(n) = \left( n - \dfrac{k(k+1)}{2}, k-n+\dfrac{k(k+1)}{2} \right)$$
\begin{enumerate}
\item Calculer $f(i)$ pour tout $i$ compris entre $0$ et $10$ et les tracer sur un quadrillage.
\item Montrer que $f$ est bien définie sur $\mathbb{N}$ (autrement dit, tout entier $n$ dans $\mathbb{N}$ a une image et il n'y a pas plusieurs images possibles pour $n$)
\item Montrer que $f$ est injective
\item En résolvant l'équation $f(n) = (p,q)$ pour $(p,q) \in \mathbb{Z}^2$, montrer que $f$ est surjective. \textit{Indication :} penser à $p+q$
\item Conclure
\end{enumerate}
\end{enumerate}	
\end{exer}


\begin{exer}{Existe t-il une bijection entre $E$ et $\mathcal{P}(E)$?}
\begin{enumerate}

Soit $E$ un ensemble. On rappelle que $\mathcal{P}(E)$ est l'ensemble des sous-ensembles de $E$. Par exemple, si $E = \{1,2,3\}$, $\mathcal{P}(E) = \{ \emptyset,\{1\}, \{2\}, \{3\}, \{1,2\}, \{1,3\}, \{2,3\}, \{1,2,3\} \} $. C'est donc un ensemble d'ensembles. \\
\item Démontrer  qu'il existe une surjection de $\mathcal{P}(E)$ sur $E$
\item  Soit $f : E \longrightarrow \mathcal{P}(E)$ une application et $A = \{x \in E | x \notin f(x) \}$.
Prouver que $A \notin f(E)$ \\
\textit{Indication} : On pourra raisonner par l'absurde en considérant un antécédent $x$  de $A$ et en discutant selon que $x \in A$ ou $x \notin A$.
\item Est-ce qu'il peut exister une bijection entre $E$ et $\mathcal{P}(E)$?
\end{enumerate}
\end{exer}
