\begin{exer}\textit{(Sinus itéré)}\quad\\[0.2cm]
On défini la suite récurrente $(u_n)_{n\geq0}$ associée à la fonction \emph{sinus} par la relation: 
$$\left\{\begin{array}{l}
u_{n+1}=\sin(u_n), \quad \forall n\in\N\\
u_0\in \ ]0,\frac{\pi}{2}]
\end{array}\right.$$
\begin{enumerate}
	\item  Montrer brièvement que la suite $u$ est strictement positive et converge vers $0$.
	\item Soient $ n$ un entier naturel et $ \alpha$ un réel. Justifier l'existence et donner les développements limités à l'ordre $n$ au voisinage de 0 de la fonction $\big(x\mapsto\sin(x)\big)$ et de la fonction $\big(x\mapsto(1+x)^\alpha\big)$
	\item 
	\begin{enumerate}
		\item Déterminer un réel $\alpha$ tel que la suite $u_{n+1}^\alpha-u_n^\alpha$ ait une limite finie non nulle.
		\item En utilisant le lemme de \textsc{Cesaro}, déterminer un équivalent simple de $u_n$.
		
	\end{enumerate}
\end{enumerate}
\end{exer}



\begin{exer}\textit{(Étude d'une suite de polynômes)}\quad\\[0.2cm]
On considère la suite de polynômes $\left(P_{n}\right)_{n \in \mathbb{N}}$ définie par $P_{0}=1$ et, pour $n \in \mathbb{N}^{*}$ :

$$
P_{n}(X)=\frac{1}{n !} X(X+n)^{n-1} .
$$

\begin{enumerate}
	\item \begin{enumerate}
		\item Expliciter les polynômes $P_{1}$ et $P_{2}$.
		\item  Donner la valeur de $P_{n}(0)$ pour tout $n \in \mathbb{N}$.
	\end{enumerate}
	\item Montrer que $$
	\forall n \in \mathbb{N}^{*}, \forall x \in \mathbb{R}, \quad P_{n}^{\prime}(x)=P_{n-1}(x+1) .
	$$
	\item En déduire que
	
	$$
	\forall n \in \mathbb{N}, \quad \forall(x, y) \in \mathbb{R}^{2}, \quad P_{n}(x+y)=\sum_{k=0}^{n} P_{k}(x) P_{n-k}(y)
	$$
	\item 
	\begin{enumerate}
		\item Montrer que pour $x \in \mathbb{R}, \quad(x+n)^{n-1} \underset{n \rightarrow+\infty}{\sim} e^{x} n^{n-1}$.
		\item Rappeler la formule de Stirling puis montrer que, pour $a \in \mathbb{R}$ et $x \in \mathbb{R}^{*}$ fixés, la suite $\left(P_{n}(x) a^{n}\right)_{n \geq 0}$ converge si et seulement si $|a| \leq \frac{1}{e}$.
	\end{enumerate}
\end{enumerate}

\end{exer}

\section*{Problème: Inégalité polynomiale de Bernstein et applications}

\begin{itemize}[label=\textbullet]
	\item Si $n \in \mathbb{N}$, on note $\mathbb{C}_{n}[X]$ le $\mathbb{C}$-espace vectoriel des polynômes à coefficients complexes de degré inférieur ou égal à $n$;
	\item Si $n \in \mathbb{N}^{*}$, on note $\mathcal{S}_{n}$ le $\mathbb{C}$-espace vectoriel des fonctions $f: \mathbb{R} \rightarrow \mathbb{C}$ vérifiant\\[-0.5cm]
	
	$$
	\exists\left(a_{0}, \ldots, a_{n}\right) \in \mathbb{C}^{n+1}, \quad \exists\left(b_{1}, \ldots, b_{n}\right) \in \mathbb{C}^{n}, \quad \forall t \in \mathbb{R}, \quad f(t)=a_{0}+\sum_{k=1}^{n}\left(a_{k} \cos (k t)+b_{k} \sin (k t)\right) .
	$$
	
	On remarque que les éléments de $\mathcal{S}_{n}$ sont des fonctions bornées ;
	\item si $I$ est un intervalle non vide de $\mathbb{R}$ et si $f$ est une fonction bornée de $I$ dans $\mathbb{C}$, on note
	
	$$
	\|f\|_{L^{\infty(I)}}=\sup _{x \in I}|f(x)| .
	$$
\end{itemize}


L'application $f \mapsto\|f\|_{L^{\infty}(I)}$ définit une norme sur le $\mathbb{C}$-espace vectoriel des fonctions bornées de $I$ dans $\mathbb{C}$ noté $\mathcal B \left(I,\C\right)$. C'est à dire que l'on a: $\forall f,g \in \mathcal B \left(I,\C\right)$, \ $\forall\lambda\in\C$:

\begin{itemize}[label=$\square$]
	\item À valeurs positives finies : $0\leq \|f\|_{L^{\infty}(I)} < +\infty$
	\item Séparabilité : $ \|f\|_{L^{\infty}(I)} = 0$ \ si et seulement si \  $f=0_{\mathcal B \left(I,\C\right)}$
	\item Homogénéité : $ \|\lambda f\|_{L^{\infty}(I)} = |\lambda|\| f\|_{L^{\infty}(I)}$ 
	\item Inégalité triangulaire : $ \|f + g \|_{L^{\infty}(I)} \ \leq \  \|f \|_{L^{\infty}(I)} +\|g \|_{L^{\infty}(I)} $
\end{itemize}



\section{Polynômes de Tchebychev}

On définit la suite de polynômes \ $ \left(T_{n}\right)_{n \in \mathbb{N}}$ \ par \ $T_{0}=1, T_{1}=X$ \ et \ $\forall n \in \mathbb{N}, T_{n+2}=2 X T_{n+1}-T_{n}$.

\begin{enumerate}
	\item Pour tout $n$ dans $\mathbb{N}$, déterminer le degré de $T_{n}$ et le coefficient dominant associé.
	\item Montrer que, pour tout $n \in \mathbb{N}$ et $\theta \in \mathbb{R},\ T_{n}(\cos \theta)=\cos (n \theta)$.
	\item  Justifier que pour tout $n \in \mathbb{N}$, la fonction de $\mathbb{R}$ dans $\mathbb{C}, \theta \mapsto T_n(\cos \theta)$ est dans $\mathcal{S}_{n}$.
	
	
	\item Pour $n \in \mathbb{N}$, calculer $\left\|T_{n}\right\|_{L^{\infty}([-1,1])}$.
	
	\item Montrer que pour tout $n \in \mathbb{N}, \ \left\|T_{n}^{\prime}\right\|_{L^{\infty}([-1,1])}=n^{2}$.
	
	\noindent On pourra commencer par établir que, pour tout $n \in \mathbb{N}$ et $\theta \in \mathbb{R}, \ |\sin (n \theta)| \leqslant n|\sin \theta|$.
\end{enumerate}


\section{Inégalité de Bernstein}

Soit $n$ un entier naturel non nul.

\begin{enumerate}[start=6]
	\item Soit $A \in \mathbb{C}_{2 n}[X]$, scindé à racines simples, et $\left(\alpha_{1}, \ldots, \alpha_{2 n}\right)$ ses racines. Montrer que
	

	\begin{equation}\tag{I.1}
	\forall B \in \mathbb{C}_{2 n-1}[X], \quad B(X)=\sum_{k=1}^{2 n} B\left(\alpha_{k}\right) \frac{A(X)}{\left(X-\alpha_{k}\right) A^{\prime}\left(\alpha_{k}\right)} .
\end{equation}
	
	Soit $P$ dans $\mathbb{C}_{2 n}[X]$, et, pour tout $\lambda \in \mathbb{C}, P_{\lambda}(X)=P(\lambda X)-P(\lambda)$.
	\item Si $\lambda \in \mathbb{C}$, vérifier que $X-1$ divise $P_{\lambda}$.
	
	Pour tout $\lambda$ dans $\mathbb{C}$, on note $Q_{\lambda}$ le quotient de $P_{\lambda} \operatorname{par} X-1$ :
	
	$$
	Q_{\lambda}(X)=\frac{P(\lambda X)-P(\lambda)}{X-1} \in \mathbb{C}_{2 n-1}[X] .
	$$
	\item Montrer que, pour tout $\lambda$ dans $\mathbb{C},\ Q_{\lambda}(1)=\lambda P^{\prime}(\lambda)$.
	
	On considère le polynôme $R(X)=X^{2 n}+1$. Pour $k$ dans $\llbracket1,2 n\rrbracket $, on note $\varphi_{k}=\frac{\pi}{2 n}+\frac{k \pi}{n}$ et $\omega_{k}=\mathrm{e}^{\mathrm{i} \varphi_{k}}$.
	
	\item Montrer que
	
	$$
	R(X)=\prod_{k=1}^{2 n}\left(X-\omega_{k}\right) .
	$$
	\item À l'aide de la formule (I.1), montrer que
	
	$$
	\forall \lambda \in \mathbb{C}, \quad Q_{\lambda}(X)=-\frac{1}{2 n} \sum_{k=1}^{2 n} \frac{P\left(\lambda \omega_{k}\right)-P(\lambda)}{\omega_{k}-1} \frac{X^{2 n}+1}{X-\omega_{k}} \omega_{k}
	$$
	
	puis en déduire que
	\begin{equation}\tag{I.2}
	\forall \lambda \in \mathbb{C}, \quad \lambda P^{\prime}(\lambda)=\frac{1}{2 n} \sum_{k=1}^{2 n} P\left(\lambda \omega_{k}\right) \frac{2 \omega_{k}}{\left(1-\omega_{k}\right)^{2}}-\frac{P(\lambda)}{2 n} \sum_{k=1}^{2 n} \frac{2 \omega_{k}}{\left(1-\omega_{k}\right)^{2}} .	
	\end{equation}

	\item Montrer que
	
	$$
	\forall \lambda \in \mathbb{C}, \quad \lambda P^{\prime}(\lambda)=\frac{1}{2 n} \sum_{k=1}^{2 n} P\left(\lambda \omega_{k}\right) \frac{2 \omega_{k}}{\left(1-\omega_{k}\right)^{2}}+n P(\lambda) .
	$$
	
	\emph{On pourra appliquer l'égalité (I.2) au polynôme $X^{2 n}$.}
	
	Soit maintenant $f$ dans $\mathcal{S}_{n}$.
	\item Montrer qu'il existe $U \in \mathbb{C}_{2 n}[X]$ tel que, pour tout $\theta \in \mathbb{R}, \ f(\theta)=\mathrm{e}^{-\mathrm{i} n \theta} U\left(\mathrm{e}^{\mathrm{i} \theta}\right)$.
	\item Vérifier que, pour tout $k \in \llbracket 1,2 n\rrbracket, \frac{2 \omega_{k}}{\left(1-\omega_{k}\right)^{2}}=\frac{-1}{2 \sin \left(\varphi_{k} / 2\right)^{2}}$ et déduire des questions 11 et 12 que
	\begin{equation}\tag{I.3}
		\forall \theta \in \mathbb{R}, \quad f^{\prime}(\theta)=\frac{1}{2 n} \sum_{k=1}^{2 n} f\left(\theta+\varphi_{k}\right) \frac{(-1)^{k}}{2 \sin \left(\varphi_{k} / 2\right)^{2}} .
	\end{equation}

	\item En déduire que
	
	\begin{equation}\tag{I.4}
	\forall \theta \in \mathbb{R}, \quad\left|f^{\prime}(\theta)\right| \leqslant n\|f\|_{L^{\infty}(\mathbb{R})} .
	\end{equation}
\end{enumerate}

\section{Quelques conséquences de l'inégalité (I.4)}

Soit $n$ un entier naturel non nul.

\begin{enumerate}[start=15]
	\item On admet que pour tout $n \in \mathbb{N}$ et $P \in \mathbb{C}_{n}[X]$, la fonction de $\mathbb{R}$ dans $\mathbb{C},\ \big(\theta \mapsto P(\cos \theta)\big)$ est dans $\mathcal{S}_{n}$. Déduire de ce qui précède et de la question 14 que
	
	$$
	\forall P \in \mathbb{C}_{n}[X], \quad \forall x \in[-1,1], \quad\left|P^{\prime}(x) \sqrt{1-x^{2}}\right| \leqslant n\|P\|_{L^{\infty}([-1,1])} .
	$$
	\item Montrer que
	
	$$
	\forall Q \in \mathbb{C}_{n-1}[X], \quad|Q(1)| \leqslant n \sup _{-1 \leqslant x \leqslant 1}\left|Q(x) \sqrt{1-x^{2}}\right| .
	$$
	
	On pourra considérer $f: \theta \mapsto Q(\cos \theta) \sin \theta$ \  et vérifier que $f \in \mathcal{S}_{n}$.
	\item Soit $R \in \mathbb{C}_{n-1}[X]$ et $t \in[-1,1]$. Montrer que
	
	$$
	|R(t)| \leqslant n \sup _{-1 \leqslant x \leqslant 1}\left|R(x) \sqrt{1-x^{2}}\right| .
	$$
	
	On pourra considérer le polynôme $S_{t}(X)=R(t X)$. 
	\item En déduire que, pour tout $P$ dans $\mathbb{C}_{n}[X]$
	
	$$
	\left\|P^{\prime}\right\|_{L^{\infty}([-1,1])} \leqslant n^{2}\|P\|_{L^{\infty}([-1,1])} .
	$$
	\item Peut-il y avoir égalité dans l'inégalité précédente?  
\end{enumerate}
