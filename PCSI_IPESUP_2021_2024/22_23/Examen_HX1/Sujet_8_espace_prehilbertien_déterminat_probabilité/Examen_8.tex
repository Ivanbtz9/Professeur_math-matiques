\begin{exer}\textit{(Polynômes de Legendre)}\quad\\[0.2cm]%p300 DM Poulet 

	
	Dans tout l'exercice, $n$ désigne un entier naturel, $\mathbb{R}[X]$ l'espace vectoriel des polynômes à coefficients réels et $\mathbb{R}_{n}[X]$ l'ensemble des polynômes réels de degré inférieur ou égal à $n$. On identifiera polynômes et fonctions polynomiales associées. Pour tout $k \in \mathbb{N}$, on note $P^{(k)}$ la dérivée $k$-ème du polynôme $P$.
	
	Pour tout $n \in \mathbb{N}$, on considère les polynômes définis par :
	
	$$
	U_{n}=\left(X^{2}-1\right)^{n} \text { et } L_{n}=\frac{1}{2^{n} n !} U_{n}^{(n)}
	$$
	
	La famille $\left(L_{n}\right)$ est appelée la famille des polynômes de Legendre.
	
	Pour tout polynôme $P$, on note $\mathscr{L}(P)$ le polynôme :
	
	$$
	\mathscr{L}(P)=\left[\left(X^{2}-1\right) P^{\prime}\right]^{\prime}
	$$
	\newpage
	

	
	\subsection{ Préliminaires}
	\begin{enumerate}
		\item \begin{enumerate}
			\item Calculer $L_{0}, L_{1}, L_{2}$ et $L_{3}$.
			\item Pour tout $n \in \mathbb{N}$, déterminer le degré et le coefficient dominant de $L_{n}$.
			\item En déduire que la famille $\left(L_{0}, \ldots, L_{n}\right)$ est une base de $\mathbb{R}_{n}[X]$.
		\end{enumerate} 
	\item Montrer que $L_{2 n}$ (respectivement $L_{2 n+1}$ ) est une fonction paire (respectivement impaire).
	\item \begin{enumerate}
		\item Montrer que pour tout $\disp n \in \mathbb{N}: L_{n}=\frac{1}{2^{n}} \sum_{k=0}^{n}\binom{n}{k}^{2}(X-1)^{k}(X+1)^{n-k}$.
		\item En déduire les valeurs de $L_{n}(-1)$ et de $L_{n}(1)$. 
	\end{enumerate}	
\begin{enumerate}
	\item Montrer que pour tout $n \in \mathbb{N}$,
	
	$$
	\begin{gathered}
	U_{n+1}^{\prime}-2(n+1) X U_{n}=0 \\
	\left(X^{2}-1\right) U_{n}^{\prime}-2 n X U_{n}=0
	\end{gathered}
	$$
	\item En dérivant les égalités polynomiales précédentes, montrer que la suite $\left(L_{n}\right)$ vérifie :
	
	$$
	\begin{gathered}
	L_{n+1}^{\prime}=X L_{n}^{\prime}+(n+1) L_{n} \\
	\mathscr{L}\left(L_{n}\right)=n(n+1) L_{n}
	\end{gathered}
	$$
	
	\item En déduire que la restriction de $\mathscr{L}$ à $\mathbb{R}_{n}[X]$ est un endomorphisme que nous noterons $\mathscr{L}_{n}$.
	
	Exprimer la matrice de $\mathscr{L}_{n}$ dans la base $\left(L_{0}, \ldots, L_{n}\right)$.
\end{enumerate}
	\end{enumerate}

	
	\subsection{Étude d'un produit scalaire et d'une base orthogonale}
	
	Pour tous $P, Q \in \mathbb{R}[X]$, on pose: $$\langle P, Q\rangle=\int_{-1}^{1} P(x) Q(x) d x$$
	
	\begin{enumerate}
		\setcounter{enumi}{3}
		\item Montrer que $\langle\cdot,\cdot\rangle$ est un produit scalaire sur $\mathbb{R}[X]$. On notera $\| \cdot\|$ la norme euclidienne associée.
		\item Montrer que pour tous $P, Q \in \mathbb{R}[X]:\langle\mathscr{L}(P), Q\rangle=\langle P, \mathscr{L}(Q)\rangle$. On dit que $\mathscr{L}$ est un endomorphisme auto-adjoint.
		\item \begin{enumerate}
			\item Montrer que pour tout $m \in \mathbb{N}$, la famille $\left(L_{n}\right)_{n \in \llbracket 0, m \rrbracket}$ est une base orthogonale de $\mathbb{R}_{m}[X]$.
			\item Montrer que pour tout $n \in \mathbb{N}, L_{n+1} \in \mathbb{R}_{n}[X]^{\perp}$.
			
		\end{enumerate}
	\item Montrer que $\left\|L_{n}\right\|^{2}=\frac{2}{2 n+1}$.
	\end{enumerate}
	
	\subsection{ Deux propriétés supplémentaires}
	\begin{enumerate}
		\setcounter{enumi}{7}
		\item En considérant un polynôme $\disp Q=\prod_{i=1}^{k}\left(X-a_{i}\right)$ de $\mathbb{R}_{n}[X]$, montrer que $L_{n+1}$ possède $n+1$ racines réelles distinctes, toutes dans l'intervalle $] - 1, 1[$.
		\item Calculer la distance de $X^{n+1}$ au sous-espace vectoriel $\mathbb{R}_{n}[X]$.
	\end{enumerate}
\hfill\\[1.5cm]

\end{exer}


\begin{exer}\textit{(Temps d'arrêt)}\quad\\[0.2cm]%p107 DS Poulet

	
	Pour tout entier $n$ tel que $n \geqslant 2$, on considère une urne $U_{n}$ contenant $n$ boules numérotées de 1 à $n$. On effectue, dans $U_{n}$, des tirages indépendants d'une boule avec remise. On suppose que tous les tirages dans $U_{n}$ sont équiprobables. On s'arrête dès que l'on obtient une boule déjà obtenue.
	
	On suppose l'expérience modélisée par un espace probabilisé fini $(\Omega, P)$ et on note $T_{n}$ la variable aléatoire égale au nombre de tirages effectués.
	\newpage 
	
	\begin{enumerate}
		\item 	Déterminer $T_{n}(\Omega)$ et justifier que :
		$$
		P\left(T_{n}>n+1\right)=0
		$$
		\item Prouver que pour tout entier naturel $k$ tel que $k \leqslant n$, on a :
		
		$$
		P\left(T_{n}>k\right)=\frac{n !}{(n-k) ! n^{k}}
		$$
	\end{enumerate}

	Pour tout $n \geqslant 2$, on considère la variable aléatoire $Y_{n}=\dfrac{T_{n}}{\sqrt{n}}$. Soit $\left.y \in\right] 0,+\infty[$ fixé.

Pour tout entier $n \geqslant 2$, on note $k_{n}(y)$ l'entier naturel égal à la partie entière de $y \sqrt{n}$ de sorte que l'on a :
	$$
	k_{n}(y) \leqslant y \sqrt{n}<1+k_{n}(y)
	$$
	
	
		\begin{enumerate}
		\setcounter{enumi}{2}
		\item Justifier que pour tout entier $n \geqslant 2$ :
		
		$$
		P\left(Y_{n}>y\right)=P\left(T_{n}>k_{n}(y)\right)
		$$
		\item Rappeler la formule de Stirling et montrer que:
		
		$$
		P\left(Y_{n}>y\right) \underset{n \rightarrow+\infty}{\sim} e^{-k_{n}(y)}\left(1-\frac{k_{n}(y)}{n}\right)^{k_{n}(y)-n}
		$$
		\item\begin{enumerate}
			\item Déterminer le développement limité à l'ordre 2 au voisinage de 0 de la fonction $f: t\longmapsto-t+(t-1) \ln (1-t)$.
			\item En déduire que:
			
			$$
			\lim _{n \rightarrow+\infty}\left(-k_{n}(y)+\left(k_{n}(y)-n\right) \ln \left(1-\frac{k_{n}(y)}{n}\right)\right)=-\frac{y^{2}}{2}
			$$
			
	\end{enumerate}
	\item Montrer que:
	
	$$
	\lim _{n \rightarrow+\infty} P\left(Y_{n} \leqslant y\right)=1-e^{-\frac{y^{2}}{2}}
	$$
\end{enumerate}

\end{exer}

\begin{exer}\textit{(Matrices symplectiques)}\quad\\[0.2cm]%p274 DM Poulet

	Dans tout le problème, $n$ désigne un entier naturel non nul : $n \in \mathbb{N}^{*}$.
	\begin{itemize}
		\item Dans $\mathscr{E}_{n}=\mathcal{M}_{n, 1}(\mathbb{R})$ espace vectoriel réel de dimension $n$, on utilisera le produit scalaire canonique défini par :
		
		$$
		\forall U, V \in \mathscr{E}_{n}, \ (U \mid V)=U^{\top} V
		$$
		\item On notera $\mathscr{M}_{n}=\mathcal{M}_{n}(\mathbb{R})$, l'espace vectoriel des matrices carrées de taille $n$ à coefficients réels.
		\item Dans $\mathcal{M}_{n}$, on notera $0_{n}$ la matrice nulle et $I_{n}$ la matrice unité. Le déterminant est noté det.
		\item $\mathscr{G}_{n}=G L_{n}(\mathbb{R})=\left\{M \in \mathscr{M}_{n}\right.$, $\left.\operatorname{det}(M) \neq 0\right\}$ désigne le groupe linéaire des matrices inversibles de $\mathscr{M}_{n}$.
		
		\item On sera enfin amené à utiliser des décompositions par blocs. On rappelle en particulier que si $A, B, C, D, A^{\prime}, B^{\prime}$, $C^{\prime}, D^{\prime} \in \mathcal{M}_{n}$ on a alors dans $\mathcal{M}_{2 n}$ :
			$$
		\begin{gathered}
		\left(\begin{array}{ll}
		A & B \\
		C & D
		\end{array}\right)\left(\begin{array}{ll}
		A^{\prime} & B^{\prime} \\
		C^{\prime} & D^{\prime}
		\end{array}\right)=\left(\begin{array}{cc}
		A A^{\prime}+B C^{\prime} & A B^{\prime}+B D^{\prime} \\
		C A^{\prime}+D C^{\prime} & C B^{\prime}+D D^{\prime}
		\end{array}\right) \\
		\operatorname{det}\left(\begin{array}{cc}
		A & C \\
		0_{n} & D
		\end{array}\right)=\operatorname{det}\left(\begin{array}{cc}
		A & 0_{n} \\
		C & D
		\end{array}\right)=\operatorname{det}(A) \operatorname{det}(D)
		\end{gathered}
		$$
	\end{itemize}


	
	\subsection{Propriétés sur les matrices symplectiques}
	
	Soit $n \in \mathbb{N}^{*}$ et soit $J_{n}$ ou simplement $J$ la matrice de $\mathscr{M}_{2 n}$ définie par :
	
	$$
	J=\left(\begin{array}{cc}
	0_{n} & -I_{n} \\
	I_{n} & 0_{n}
	\end{array}\right)
	$$
	
	On note:
	
	$$
	\mathscr{S}p_{2n}=\left\{M \in \mathscr{M}_{2 n}, M^{\top} J M=J\right\}
	$$
	\begin{enumerate}
		\item Calculer $J^{2}$ et $J^{\top}$ en fonction de $I_{2 n}$ et $J$. Montrer que $J$ est inversible et identifier son inverse.
		\item Vérifier que $J \in \mathscr{S}p_{2n}$ et que pour tout réel $\alpha$,
		
		$$
		K(\alpha)=\left(\begin{array}{cc}
		I_{n} & 0_{n} \\
		-\alpha I_{n} & I_{n}
		\end{array}\right) \in \mathscr{S}p_{2n}
		$$
		\item Pour tout $U \in \mathscr{G}_{n}$, vérifier que $L_{U}=\left(\begin{array}{cc}U & 0_{n} \\ 0_{n} & \left(U^{-1}\right)^{\top}\end{array}\right)$ est dans $\mathscr{S}p_{2n}$.
		\item Si $M \in \mathscr{S}p_{2n}$, préciser les valeurs possibles $\operatorname{de} \operatorname{det}(M)$.
		\item Montrer que le produit de deux éléments de $\mathscr{S}p_{2n}$ est un élément de $\mathscr{S}p_{2n}$.
		\item Montrer qu'un élément de $\mathscr{S}p_{2n}$ est inversible et que son inverse appartient à $\mathscr{S}p_{2n}$.
		\item Montrer que si $M \in \mathscr{S}p_{2n}$ alors $M^{\top} \in \mathscr{S}p_{2n}$.
		
		Soit $M$ une matrice de $\mathscr{M}_{2 n}$ écrite sous la forme:
		
		$$
		M=\left(\begin{array}{ll}
		A & B \\
		C & D
		\end{array}\right) \text { avec } A, B, C, D \in \mathcal{M}_{n}
		$$
		\item Déterminer les relations sur $A, B, C, D$ caractérisant l'appartenance de $M$ à $\mathscr{S}p_{2n}$. 
	\end{enumerate}

\subsection{Centre de $\mathscr{S}p_{2n}$}


	

	
	On s'intéresse ici au centre $\mathcal{Z}$ de $\mathscr{S}p_{2n}$ c'est-à-dire: $\mathcal{Z}=\left\{M \in \mathscr{S}p_{2n}, \forall N \in \mathscr{S}p_{2n}, M N=N M\right\}$.
	
	\begin{enumerate}
		\setcounter{enumi}{8}
		\item Justifier l'inclusion suivante $:\left\{-I_{2 n}, I_{2 n}\right\} \subset \mathcal{Z}$.
		
		Réciproquement, soit $M \in \mathcal{Z}$ écrite sous la forme :
		
		$$
		M=\left(\begin{array}{ll}
		A & B \\
		C & D
		\end{array}\right) \text { avec } A, B, C, D \in \mathscr{M}_{n}
		$$
		\item En utilisant $L=\left(\begin{array}{cc}I_{n} & I_{n} \\ 0_{n} & I_{n}\end{array}\right)$ et sa transposée, obtenir $B=C=0_{n}$ et $D=A, A$ étant inversible.
		\item Soit $U \in \mathscr{G}_{n}$. En utilisant $L_{U}=\left(\begin{array}{cc}U & 0_{n} \\ 0_{n} & \left(U^{-1}\right)^{\top}\end{array}\right)$, montrer que $A$ commute avec toute matrice $U \in \mathscr{G}_{n}$.
		\item Conclure que $A \in\left\{-I_{n}, I_{n}\right\}$ et $\mathcal{Z}=\left\{-I_{2 n}, I_{2 n}\right\}$.
		
		Indication : on montrera d'abord que les matrices $I_{n}+E_{i, j}$ commutent avec $A$, où $\left(E_{i, j}, 1 \leqslant i, j \leqslant n\right)$ est la base canonique de $\mathscr{M}_{n}$.
	\end{enumerate}
\end{exer}
