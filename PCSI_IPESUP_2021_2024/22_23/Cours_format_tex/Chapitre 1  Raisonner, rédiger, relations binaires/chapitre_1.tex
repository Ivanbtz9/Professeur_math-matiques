\section{Rédaction et quelques types de raisonnements}
\begin{itemize}[\Large$\bullet$]
	\item Une copie est la seule chose que le correcteur connaîtra de vous lors d'une épreuve écrite, votre note dépendra seulement de celle-ci. Il tentera d'évaluer plusieurs éléments : votre compréhension
	du sujet, vos connaissances mathématiques, vos capacités à exposer ces connaissances, votre maîtrise de la langue
	française et enfin le soin dont vous saurez faire preuve.
	\item Les exemples introduits par le sigle $\happy$ sont des exemples où la rédaction est correcte.
	\item Ceux introduits par $\sad$ sont des exemples où la rédaction est incorrecte et sanctionnée.
\end{itemize}



\subsection{Introduire tous les objets utilisés}
\begin{itemize}[\Large$\bullet$]
	\item On présente toujours les nouveaux venus, tous les objets que l'on introduit dans la rédaction doivent être définis au préalable. Introduire un nouvel objet quelconque se fait simplement avec le mot « soit » \\ou « soient » quand il y en a plusieurs.\\ 

\noindent $\sad$ $x\geq0$. Si $x=1$ alors vrai, sinon faux. \\
\noindent $\happy$ Soit $x\in\R^+$. Si $x=1$ la proposition $\Ps$ est vraie sinon elle est fausse. \\[0.2cm]

\noindent $\sad$ Cherchons les racines du polynôme $X^2 + X - 1$. $\Delta = 5$ donc les racines sont $\dfrac{-1+\sqrt{5}}{2}$ et $\dfrac{-1-\sqrt{5}}{2}$ .\\
\noindent $\happy$ On note $\Delta$ le discriminant du polynôme $X^2 + X - 1$. On a $\Delta = 5$ donc. . . 
\end{itemize}

\subsubsection{Quantification}
\begin{itemize}[$\bullet$]
	\item Quand vous énoncez une propriété en lien avec des objets mathématiques, il faut impérativement préciser si cette propriété est vérifiée par tous les objets de ce type, par une certaine partie de ces objets, ou par au moins un de ces objets. Pour cela on utilise les quantificateurs:\\
	
	\text{"Pour tout" :} {\Large$\forall$}\quad\quad\quad\quad
	\text{"Il existe" :} {\Large$\exists$}\quad\quad\quad\quad
	\text{"Il existe un unique" :} {\Large$\exists!$}\quad\quad\quad\quad
	\text{"Il n'existe pas" :} {\Large$\nexists$}\\
	
	
	\noindent\danger\quad Utiliser un quantificateur n'introduit pas un objet. Par exemple si vous écrivez: \ $\exists x \in \R \ | \  x > 0$\\ Alors $x$ n'est pas
	introduit pour autant, et vous ne pouvez pas utiliser la lettre $x$ directement après :\\[0.2cm]
	
	\noindent $\sad$ $\exists x \in \R \ | \  x > 1$. Alors $x\geq 0$. \\
	\noindent $\happy$ $\exists x \in \R \ | \  x > 1$. Soit $x\in\R$ tel que $ x > 1$, alors $x\geq 0$.
\end{itemize}

\textbf{Point Méthode à la rédaction:}
\begin{enumerate}
	\item[\textbf{*}] Annoncer ce que l'on va faire sans recopier la question:\\ \textit{Quelques mots simples suffisent, comme : « montrons que... », « il suffit de montrer que... ».}
	\item[\textbf{**}] Tout justifier :\\
	\textit{Il faut toujours justifier chaque étape d'un raisonnement. Lancer des affirmations sans les étayer est tout simplement interprété comme une tentative de bluff.} \newpage
	\item[\textbf{***}] Utiliser les théorèmes et les propositions :\\ 
	\textit{N'oubliez jamais de vérifier toutes les hypothèses des théorèmes et propositions utilisés de manière explicite. Lorsque l'on utilise un théorème (exemple : le théorème des valeurs intermédiaires) qui possède un nom, il faut toujours le donner.}
	\item[\textbf{****}] Écrire la conclusion et la souligner :
	\textit{Précisez toujours le résultat auquel vous aboutissez et} \fbox{\textit{encadrez-le}} \textit{ou bien} $\underline{\textit{soulignez-le}}$.\\ 
\end{enumerate}

\noindent\textbf{Une articulation logique:} Une démonstration mathématique est une succession d'arguments et de calculs. Pour que tout cela ait un sens, il faut
relier ces arguments et ces calculs par l'utilisation des mots suivants : {\large\emph{d'où, alors, donc, par conséquent, ainsi, c'est-à-dire, ce qui équivaut à, ou encore, or, puisque, de plus...}}\\

\noindent\danger\quad On ne mélange pas les quantificateurs et les symboles d'implication avec la rédaction en français :\\[0.25cm]

	\noindent$\sad$ $\forall x\in \N, ~x$ est positif. \\
	\noindent$\happy$ $\forall x\in \N,~ x\geq0$.\\
	\noindent$\happy$ Pour tout $x$ dans $\N$, $x$ est positif. \\
	 \noindent$\happy$ Pour tout $x\in\N$, $x\geq0$. \\[0.25cm]

	 \noindent$\sad$ Le module de $2e^{\frac{3\pi i}{2}}$ vaut $2$ et celui de $2+i$ vaut $\sqrt{5}$ $\Rightarrow 2e^{\frac{3\pi i}{2}} \neq 2+i$.\\
	\noindent$\happy$ Le module de $2e^{\frac{3\pi i}{2}}$ vaut $2$ et celui de $2+i$ vaut $\sqrt{5}$ donc on a: $2e^{\frac{3\pi i}{2}} \neq 2+i$. \\

\subsubsection{implication et équivalence}

\noindent\fcolorbox{blue}{white}{
	\begin{minipage}{0.9\linewidth}
	\textbf{Raisonner par implications:}
	Quand on veut montrer que $p \alors q$ est vraie : 
	« On suppose p vraie et on montre ce que cela implique dans le but de montrer que q est vraie. »
	\end{minipage}}
\hfill\\[0.2cm]
\begin{ex}	
	Démontrer que: $\forall x\in[0,1]$,  $x-x^2 \in\N \alors x\in\{0,1\}$.
\end{ex}
\begin{ex}
Démontrer que: $\forall x,x'\in\R\backslash\{1\}$,  $ x\neq x' \alors \dfrac{x+1}{x-1}\neq \dfrac{x'+1}{x'-1}$.
\end{ex}
\begin{exer}
	Soit $n,k\in\N$. Montrons que si l'entier $n$ est un multiple de $k$ alors $n^2$ l'est également.
\end{exer}
\hfill\\[0.2cm]
\noindent\fcolorbox{blue}{white}{
	\begin{minipage}{0.9\linewidth}
		\textbf{Raisonner par implications:}
		Quand on veut montrer que $p \alors q$ est fausse : 
		« On suppose p vraie et on montre ce que cela implique dans le but de montrer que (non q) qui se note aussi {\Large$\neg$}$q$ est vraie. Donc que $q$ est fausse. »
\end{minipage}}\hfill\\[0.2cm]
\begin{ex}
	Démontrer qu'il est faux d'affirmer: $\forall x,y\in\R$,  $ x<y \alors \sin(x)<\sin(y)$.
\end{ex}
\begin{exer}
	Démontrer qu'il est faux d'affirmer: $\forall x \in\R$,  $ x^2+x>0 \alors x>0$.
\end{exer}

\noindent\fcolorbox{blue}{white}{
	\begin{minipage}{0.9\linewidth}
		\textbf{Raisonner par équivalence:}
		Quand on veut montrer que $p \equi q$ est vraie, deux possibilités :
		\begin{itemize}[$\bullet$]
			\item Soit on raisonne par double implication :
			\begin{itemize}[$\star$]
				\item Supposons $p$ vraie. Montrons que $q$ est vraie.
				\item Réciproquement, supposons $q$ vraie. Montrons que $p$ est vraie.
			\end{itemize}
			\item soit on raisonne directement par équivalences en changeant peu à peu $p$ en $q$ : $$ p \equi . . . . . . \equi . . . . . . \equi q$$
		\end{itemize}
		
\end{minipage}}\hfill\\[0.2cm]
\begin{ex}
	Démontrer que: $\forall x,y\in\R$,  $ x^2+y^2=0 \equi x= y= 0$.
\end{ex}
\begin{ex}
	Démontrer que: pour tout $x \in[-\pi,\pi]$, $\cos(2x) = \dfrac{1}{2} $ si et seulement si $x\in\left\{-\dfrac{5\pi}{6},~ -\dfrac{\pi}{6} ,~\dfrac{\pi}{6},~ \dfrac{5\pi}{6}\right\}$.
\end{ex}

\subsubsection{Unicité d'un objet mathématique}
\noindent\fcolorbox{blue}{white}{
	\begin{minipage}{0.9\linewidth}
		\textbf{Unicité d'un objet:} Quand on veut montrer qu'un ensemble E contient au plus un élément vérifiant une propriété $\Ps$ , on peut procéder ainsi:
		
		Soient $x, x' \in E$. Faisons l'hypothèse que $x$ et $x'$ vérifies $\Ps$ et montrons que $x = x'$.	
\end{minipage}}
\hfill\\[0.2cm]
\noindent\danger\quad Ci-dessus, il n'est pas nécessaire de supposer $x$ et $x'$ différents et de raisonner par l'absurde.
\begin{ex}
Démontrer : $\exists!x \in \R^+ \ | \ x^2 = 1$.  
\end{ex}
\begin{ex}
	Démontrer : $\forall n\in\N^* \ \exists!(p,q)\in\N^2 \ | \ n= (2p+1)2^q$.
\end{ex}
\begin{exer}
	Soit $f : \R\longrightarrow\R$ une application contractante. \\(ie: il existe $k\in [0,1[$ tel que pour tout $(x,y)\in I^2$, $|f(x)-f(y)|\leq k|x-y|)$.\\[0.2cm] 
	S'il existe $x_0\in \R$ tel que $f(x_0) =x_0$ on dit que la fonction $f$ admet un point fixe.\\[0.2cm]  
	Montrer qu'il y a unicité  de ce point fixe s'il y a existence de celui-ci.  
\end{exer}
\subsubsection{Existence d'un objet mathématique}

\noindent\fcolorbox{blue}{white}{
	\begin{minipage}{0.9\linewidth}
		\textbf{Existence d'un objet:} Quand on veut montrer qu'un ensemble E contient au moins un élément vérifiant une propriété $\Ps$ , on peut procéder ainsi:
		\begin{itemize}[$\bullet$]
			\item Faire une preuve constructive qui fait apparaître l'objet vérifiant la propriété $\Ps$. 
			\item Utiliser des propositions ou des théorèmes qui nous offrent l'existence de l'objet sans pour autant nous donner l'expression explicite de celui-ci. 
		\end{itemize} 	
\end{minipage}}
\hfill\\[0.2cm]
\begin{ex}
	Démontrer que pour tout $z\in\C^*$ il existe $z_0\in\C^*$ tel que: $|z_0|=1$ et $Arg(z)=Arg(z_0)$. 
\end{ex}

\begin{ex}
	On considère $\forall x\in\R$ $f(x)= 4x^3-3x^2+2x-1$.\\
	Démontrer que la fonction polynomiale $f$ admet une racine réelle.  
\end{ex}

\begin{exer}
	Soit $a,b\in\R$ avec $a<b$.
	
	\noindent Montrer que toute application continue $f : [a , b ] \longrightarrow[a , b ]$ admet un point fixe.
\end{exer}


\subsubsection{Inclusion et égalité d'ensembles}
\noindent\fcolorbox{blue}{white}{
	\begin{minipage}{0.9\linewidth}
	Quand on veut montrer une inclusion $E \subset F$ , on écrit: 
	Soit $x \in E$ et on montre que $x \in F$.	
\end{minipage}}
\hfill\\
\begin{ex}
Démontrer : $\{x \in \R \ | \  \exists y \in \R^+ , x > y\} \subset \R^+$.
\end{ex}

\begin{ex}
On note $2\N$ l'ensemble des entiers naturels pairs et on pose $E = \{k(k + 1) \ | \ k\in \N\}$.

\noindent Montrer que $E \subset 2\N$.  
\end{ex}

\noindent\fcolorbox{blue}{white}{
	\begin{minipage}{0.9\linewidth}
	Quand on veut montrer une égalité d'ensembles $E = F$, deux possibilités :
		\begin{itemize}[$\bullet$]
			\item Soit on raisonne par double inclusion :
			\begin{itemize}[$\star$]
				\item Soit $x \in E$. Montrons que $x \in F$ .
				\item Réciproquement, soit $x \in F$. Montrons que $x \in E$ .
			\end{itemize}
			\item Soit on raisonne directement par équivalences :
			
			 $\text{« Pour tout}\  x \in E \ :\  x \in E \equi . . . . . . \equi . . . . . . \equi x\in F.$»
		\end{itemize}\end{minipage}}\hfill\\[0.2cm]
	
\begin{ex}
	Montrer que : $\R^- = \{x \in\R \ | \  \forall y \in \R^+ , x\leq y\}$.
\end{ex}

\begin{ex}
	Soient $E$ un ensemble et $A,B$ deux sous-ensembles de E. 
	
	\noindent Pour $X\subset E$ on note $\overline{X}$ le complémentaire de $X$ dans $E$ défini par: $\overline{X} = \{x\in E \ | \ x \not\in F\}$.
	
	\noindent Montrer que : $\overline{A\cup B} = \overline{A}\cap \overline{B}$. 
\end{ex}

	
\begin{exer}
	Soit $A=\{(x,y)\in\mathbb R^2;\ 4x-y=1\}$ et $C=\{(t+1,4t+3);\ t\in\mathbb R\}$. Démontrer que $A=C$. 
\end{exer}

\subsubsection{Le raisonnement par récurrence}
\begin{enumerate}[$\bullet$]
	\item Le raisonnement par récurrence repose sur le principe suivant : 
	
	
	\centering Si $\underbrace{\Ps_0 \text{ est vraie}}_{\text{Initialisation}}$ et si $\underbrace{(\forall n \in \N, \ \Ps_n \alors \Ps_{n+1})}_{\text{Hérédité}}$ alors $(\forall n \in \N , \ \Ps_n)$.\hfill\\[0.8cm]
	
	
\raggedright
\noindent\fcolorbox{blue}{white}{
	\begin{minipage}{0.9\linewidth}
		\centering\textbf{Récurrence simple.}\\
		
	\raggedright Quand on veut montrer par récurrence que: ($\forall n \in \N \ , \  \Ps_n$), on rédige ainsi:
		\begin{itemize}[$\bullet$]
			\item Expliciter $\Ps_{n}$: Pour $n\in\N$ on note $\Ps_{n} : "\dots\dots\dots\dots"$
			\item Initialisation: Vérification que $\Ps_0$ est vraie.
			\item Hérédité: Soit $n \in \N$, supposons $\Ps_n$ vraie. Montrons que $\Ps_{n+1}$ est vraie.
			\item Conclusion: $\forall n \in \N \ , \  \Ps_n$
		\end{itemize} 	
\end{minipage}}
\hfill\\[0.2cm]
\noindent\danger \quad Supposer que pour tout $n \in \N$ , $\Ps_n$ est vraie à l'hérédité est une erreur. En effet, c'est ce que nous cherchons à démontrer. 	
\end{enumerate}
\begin{ex}
	Notons, pour tout $n \in \N^*$, $S_n$ la somme des cubes des entiers entre $1$ et $n$ et montrons par récurrence sur $\N^*$  que $S_n = \left(\dfrac{n(n+1)}{2}\right)^2$.
\end{ex}

\begin{rmq}
	Pour nier une phrase contenant un ou plusieurs quantificateurs, on réécrit cette phrase en remplaçant tous les « $\forall$ » par des « $\exists$ » et tous les « $\exists$» par des « $\forall$ », puis on nie le prédicat final.
\end{rmq}

\begin{ex}

\begin{rmq}\hfill
	
	
		\noindent  On peut toujours permuter les quantificateurs « $\forall$ » entre eux, et les quantificateurs « $\exists$ » entre eux.
\end{rmq}

\begin{ex}\hfill
	
	
	\noindent Les propositions « $\forall x \in \R^+ , \forall y \in \R^- , x > y$ » et « $\forall y \in \R^- , \forall x \in \R^+, x > y$ » sont équivalentes.
	
	
	\noindent Les propositions « $\exists x \in \R^+ , \exists y \in \R^- , x > y$ » et « $\exists y \in \R^- , \exists x \in \R^+, x > y$ » sont équivalentes.
\end{ex}

\noindent\danger\quad En revanche, la permutation d'un « $\forall$ » et d'un « $\exists$ » n'est pas automatique.
 \begin{ex}

 exemple). Mais « $\exists p \in \N, \forall n\in \N  \ | \ n \leq p$ » est une proposition fausse car aucun entier est supérieur à tout autre entier
 (l'ensemble $\N$ n'est pas majoré).
 \end{ex}

\subsubsection{Loi de composition interne}

\noindent\fcolorbox{vert}{white}{
	\begin{minipage}{1\linewidth}
		\begin{defi} Une \textit{loi de composition interne} sur un ensemble $E$ est une application de $E^2$ dans $E$.
			$$ \begin{array}{cccc}
			\mathlarger\star:& E\times E& \longrightarrow& E\\
			  & (x,y)&\longmapsto & x\star y
			\end{array}$$	
		\end{defi}
\end{minipage}}\hfill\\

\begin{ex}
	\begin{itemize}[$\bullet$]\item[]
		\item Les opérations $+$, $\times$ définissent des lois de composition interne sur les ensembles de nombres usuels $(\N, \Z, \Q, \R, \C)$.
		\item La composition $\circ$ est une loi de composition interne sur $\R^\R$. 
		\item[] 
	\end{itemize}
	 
\end{ex}

\noindent\fcolorbox{vert}{white}{
	\begin{minipage}{1\linewidth}
		\begin{defi} Soit $\star$ une loi de composition interne sur un ensemble $E $.
			\begin{enumerate}[$\bullet$]
				\item La loi $\star$ est \textit{commutative} si\\[-0.3cm]
				$$\forall (x,y)\in E^2, \quad x\star y = y \star x$$
				\item La loi $\star$ est \emph{associative} si\\[-0.3cm]
				$$\forall (x,y,z)\in E^3, \quad (x\star y) \star z = x \star (y \star z)$$
				\item La loi $\star$ admet \emph{un élément neutre} si\\[-0.3cm]
				$$\exists e \in E, \ \forall x\in E, \quad  x\star e =  e \star x = x$$
				\item Si la loi $\star$ admet un élément neutre $e\in E$, un élément $x \in E$ admet \emph{un symétrique} dans $E$ pour $\star$ si\\[-0.3cm]
				$$\exists y \in E, \quad  x\star y =  y \star x = e$$
				
			\end{enumerate}	
		\end{defi}
\end{minipage}}
\begin{ex}
	Étudier les caractéristiques des lois de l'exemple précédent.  
\end{ex}
\begin{rmq}
			\begin{enumerate}[$\bullet$]
				\item[]
				\item Toutes les lois ne sont pas commutatives: la composition des fonctions est un contre-exemple.
				\item L'associativité indique qu'il n'y a pas d'ordre de priorité dans les calculs qui n'impliquent que $\star$: on peut
				donc les mener dans l'ordre que l'on veut.
			\end{enumerate}	
\end{rmq}

\begin{rmq}Soient $A\subset\R$ et $x\in\R$.\\
	$x = sup A$ si et seulement si $x$ est le plus petit des majorants de $A$\\[0.25cm]\noindent si et seulement si \  $\left\{\begin{array}{l}x \text{ majore }A.\\ \text{Si } y \text{ majore } A \text{alors} x\leq y.
	\end{array}\right.$ si et seulement si \  $\left\{\begin{array}{l}\forall a\in A, \ a\leq x.\\ (\forall y \in \R,\  x> y) \text{ alors } y\text{ ne majore pas } A.
	\end{array}\right.$ \\[0.25cm]\noindent si et seulement si \  $\left\{\begin{array}{l}\forall a\in A, \ a\leq x.\\ \forall y \in \R,\  x> y,\ \exists a\in A,\ y<a.
	\end{array}\right.$ si et seulement si \  $\left\{\begin{array}{l}\forall a\in A, \ a\leq x.\\ \forall \epsilon>0,\ \exists a\in A,\ x - \epsilon < a \leq x.
	\end{array}\right.$
\end{rmq}

\subsection{Droite numérique achevée $\overline{\R}$}


\noindent\fcolorbox{vert}{white}{
	\begin{minipage}{1\linewidth}
		\begin{defi} L'ensemble $\overline{\R} = \R \cup \{-\infty , +\infty\}$ est appelé \emph{droite numérique achevée}.
		\end{defi}
\end{minipage}}\hfill\\[0.35cm]
\noindent On prolonge l'ordre de $\R$ à $\overline{\R}$ par : $\forall x \in \R, -\infty \leq x \leq +\infty$.

\noindent On a $sup ~\overline{\R}  = +\infty = sup ~\R$ est le plus grand élément de $\overline{\R}$ et $inf~ \overline{\R}  = -\infty = inf ~\R$ est
le plus petit élément de $\overline{\R}$.

\noindent On a ainsi que toute partie de $\overline{\R}$ admet un \emph{sup} et un \emph{inf}, éventuellement $\pm \infty$.

		\item $sup ~\emptyset = -\infty$  dans $\overline{\R}$.
		\item $inf ~]-\infty, 3] = -\infty$ dans $\overline{\R}$.
	\end{itemize}
 
\end{ex}

			\begin{enumerate}[$\bullet$]
				\item On peut prolonger l'addition $+$ sur $\R$ à $\overline{\R}$.
					\begin{enumerate}[$\square$]
					\item $\forall x \in \R \cup \{-\infty\},\ x + (- \infty) = -\infty$
