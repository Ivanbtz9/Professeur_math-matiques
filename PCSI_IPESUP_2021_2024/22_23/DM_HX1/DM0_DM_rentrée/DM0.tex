	\begin{exer}\textit{\space}\\
		Pour $n \in \N$, $f_n$ désigne la fonction de $]0, \pi]$ dans $\R$ définie par
		$$\forall x \in ]0, \pi], \quad f_n(x) = \dfrac{\sin\left(\left(n+\dfrac{1}{2}\right)x\right)}{\sin\left(\dfrac{x}{2}\right)} $$
		\begin{enumerate}
			\item (**) En utilisant la définition de la dérivée, montrer que $\lim\limits_{u \rightarrow 0} \dfrac{\sin(u)}{u} = 1$
			\item (**) En déduire que la fonction $f_n$ admet une limite que l'on précisera en $0$. \\ %\\
			Ceci permet de prolonger $f_n$ en une fonction continue sur $[0, \pi]$, encore notée $f_n$ et de poser $$I_n =\int_{0}^{\pi}f_n(t) dt$$.
			\item (***) Si $n \in \N$, calculer $I_{n+1} - I_n$ , puis $I_n$. 
		\end{enumerate}
	\end{exer}
