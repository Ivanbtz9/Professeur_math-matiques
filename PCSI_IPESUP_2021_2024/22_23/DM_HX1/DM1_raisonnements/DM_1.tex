\begin{exer}\textit{(Temps : 1 heure et 30 minutes)}\\
	
	
	\begin{enumerate}
		\item Étudier le sens de variation de la fonction $g$ définie pour tout $t$ de $\R$ par : $ g(t) = e^t - t - 1$.
		\item Quel est le minimum de la fonction $g$ sur $\R$ ?
		\item En déduire les inégalités suivantes :
		\begin{enumerate}
			\item pour tout réel $t$ :
			\begin{multicols}{3}
			\begin{itemize}
				\item[$\quad$] $ e^t \geq 1+t$ \item[$\quad$] $ e^t > t$ \item[$\quad$] $-te^{-t}>-1$
			\end{itemize}
			\end{multicols}
			\item pour tout réel $t$ tel que $t > -1$ :
			$$\ln(1 + t ) \leq t$$
			\item pour tout réel $x$ :
			$$\ln(1 - xe^{-x}) \leq -xe^{-x}$$
		\end{enumerate}
		\item On considère la fonction $f$ définie par la relation: 
		$$f(x)= x^2- 2\ln(e^x -x)$$
		\begin{enumerate}
			\item Montrer que $f$ est définie sur $\R$ et que pour tout $x \in \R$ , $f(x) = x^2 - 2x - 2\ln(1 - xe^{-x})$.
			\item En déduire la limite de $f$ en $+\infty$.
			\item Montrer que pour tout $x<0$: $f(x) = x^2\left(1-2\dfrac{\ln(-x)}{(-x)^2}\right) - 2 \ln\left(1-\dfrac{e^x}{x}\right) $
			\item En déduire la limite de $f$ en $-\infty$.
			\item Justifier que $f$ est dérivable sur $\R$ et calculer $f'$. Montrer que pour tout réel $x$ : $$f'(x) = \dfrac{2(x-1)(e^x-x-1)}{e^x-x}$$
			Dresser le tableau de variation de la fonction $f$.
			\item On considère la parabole $(\mathscr{P})$ d'équation $y = x^2 - 2x$ et $(\mathscr{C})$ la courbe représentative de $f$. Tracer dans un même repère orthonormal (unité : 3cm) l'allure des courbes $(\Ps)$ et $(\Cs)$ et les tangentes horizontales $(\Ds)$ et $(\Ds')$ à $(\Cs)$.
			\item Montrer que $(\Ps)$ et $(\Cs)$ sont asymptotes en $+\infty$, c'est-à-dire que la limite en $+\infty$ de $f(x) - x^2 -2x$ est nulle.
			Étudier les positions relatives des courbes $(\Ps)$ et $(\Cs)$.
		\end{enumerate}
		\item Soit $n$ un entier naturel, on pose :
		$$u_n = \int_{0}^{n}xe^{-x}dx $$
		\begin{enumerate}
			\item Démontrer que la suite $(u_n)_{n\geq0}$ est croissante.
			\item Calculer $u_n$ pour tout $n\in\N$.\\
			 On pourra d'abord calculer la dérivée de $h : x \longmapsto -(x + 1)e^{-x}$.
			\item Déterminer la limite de la suite $u$.
		\end{enumerate}
	\item L'aire du domaine (en unités d'aire) limité par les droites d'équation $x = 0$, $x = n$, entre la parabole $(\Ps)$ et la courbe $(\Cs)$ est définie par:
	$$I_n = -2\int_{0}^{n}\ln(1-xe^{-x})dx $$\newpage
	\begin{enumerate}
		\item Montrer en utilisant certaines des questions précédentes que $I_n \geq 2u_n$ .
		\item On admet que la suite $(I_n)$ converge vers une limite $l$. Montrer que : $l\geq2$.
	\end{enumerate}
	 
	\end{enumerate}
\end{exer}

\begin{exer}\textit{(Temps : 1 heure et 20 minutes)}\\
	On définit la suite de Fibonacci $(F_n )_{n\in\N}$ par $F_0 = 0$, $F_1 = 1$ et la relation de récurrence : $$\forall n\in\N , \  F_{n+2} = F_{n+1} + F_n$$
	\begin{enumerate}
		\item Déterminer la liste des $10$ premiers nombres de Fibonacci (de $F_1$ à $F_{10}$ ).
		\item Montrer que : $\forall n\in \N , n\geq 6 \alors  F_n > n$.\\
		Que peut-on en déduire pour la suite $(F_n )_{n\in\N}$?
		\item Montrer que : $\forall n\in \N, \ n\geq 2 \alors F_n < F_{n+1} \leq 2F_n$. 
		\item Montrer que : $\forall n\in \N^*,\sum\limits_{k=1}^{n}F_k = F_{n+2}-1$
		\item Montrer que : $\forall n\in \N^*,\sum\limits_{k=1}^{n}F_k^2 = F_nF_{n+1}$
		\item Montrer que : $\forall n\in \N^*,\sum\limits_{k=1}^{2n-1}F_kF_{k+1} =F_{2n}^2$
		\item Montrer que : $\forall n\in \N^*,F_{2n} = F_{n+1}^2-F_{n-1}^2$ et $F_{2n+1}= F_{n+1}^2+ F_n^2$.
		\item Montrer que : $\forall n\in \N^*,\sum\limits_{k=0}^{n} \begin{pmatrix} 2n-k\\k\end{pmatrix} = F_{2n+1}$ et $\sum\limits_{k=0}^{n} \begin{pmatrix} 2n+1-k\\k\end{pmatrix} = F_{2n+2}$
		\item Montrer que si l'on pose: $\varphi = \dfrac{1+\sqrt{5}}{2}$ : $\forall n\in \N^*, F_{n-1}+ \varphi F_n = \varphi^n.$
		\item  Montrer que si l'on pose: $\overline{\varphi} = \dfrac{1-\sqrt{5}}{2}$ : $\forall n\in \N, F_n = \dfrac{\varphi^n - \overline{\varphi}^n}{\varphi- \overline{\varphi}}.$
	\end{enumerate}
\end{exer}

\begin{exer}\textit{(Temps : 1 heure et 10 minutes)}\\
	
	\begin{enumerate}
		\item Résoudre dans $\R$ :
		\begin{enumerate}
			\item $\sqrt{x-9} +\sqrt{x-24} = x$
			\item $x-1 < \sqrt{x^2-2}$
			\item  $ |(x - 3)(x - 5)|  > x - 3$
			\item  $ |x| + |x-1| +  |x-2| \leq 6$
		\end{enumerate}
		\item En raisonnant par \textbf{équivalences}, résoudre les inéquations :
		\begin{enumerate}
			\item $|1-x| \leq \dfrac{x}{2-x} $
			\item $x+1 \leq \sqrt{x+2}$
		\end{enumerate}
	\item \begin{enumerate}
		\item Montrer que toute fonction $f : \R \longrightarrow \R$ peut s'écrire comme la somme d'une fonction polynomiale de degré	au plus $2$ et d'une fonction s'annulant en $-1$, $0$ et $1$. Y a-t-il unicité ?
		\item Montrer que toute fonction continue $f : [0,1] \longrightarrow \R$ est la somme d?une fonction linéaire $(x \longmapsto ax)$ et d'une fonction d'intégrale nulle sur $[0,1]$ . Y a-t-il unicité ?
	\end{enumerate}
		 
	\end{enumerate}
\end{exer}
