\section*{Dans tout le problème :}
\begin{itemize}
	\item Pour tout entier naturel $n$, on note $\R_{n}[X]$ l'espace vectoriel des polynômes à coefficients réels de degré inférieur ou égal à $n$
	\item On identifie le polynôme $\disp P=\sum_{k=0}^{n} \lambda_{k} X^{k}$ de $\mathbb{R}_{n}[X]$ avec la fonction polynomiale $\disp x \mapsto \sum_{k=0}^{n} \lambda_{k} x^{k}$, \\avec la convention $0^{0}=1$
	%\item On rappelle la formule de Stirling : $n!$  est équivalent à $\disp n^{n} \mathrm{e}^{-n} \sqrt{2 \pi n}$ lorsque l'entier $n$ tend vers $+\infty$. Elle ne sera pas utilisée dans ce problème, en revanche vous devez la connaître sans hésitation...
\end{itemize}




\textit{Dans la partie I, on étudie le cas des polynômes de Bernstein. La partie II est consacrée aux polynômes d'interpolation de Lagrange.}

\section*{Partie I : Quelques propriétés des polynômes de Bernstein}

Pour tout entier $n \in \mathbb{N}^{*}$ et tout entier $k \in \llbracket 0, n \rrbracket$, on note $B_{n, k}$ le polynôme de $\mathbb{R}_{n}[X]$ défini par :

$$
B_{n, k}(X)=\binom{n}{k} X^{k}(1-X)^{n-k}
$$

On pose pour tout $k \in \llbracket 0, n \rrbracket,\ A_{k}=X^{k}$ et on note $\disp \mathcal{C}_{n}=\left(A_{0}, A_{1}, \ldots, A_{n}\right)$ la base canonique de $\mathbb{R}_{n}[X]$.

Soit $T_{n}$ l'application définie sur $\mathbb{R}_{n}[X]$ telle que: $$ \forall P \in \mathbb{R}_{n}[X],\ \left(T_{n}(P)\right)(X)=\sum_{k=0}^{n} P\left(\frac{k}{n}\right) B_{n, k}(X)$$

\begin{enumerate}
	
	\item Dans cette question uniquement, on choisit $n=2$.
	\begin{enumerate}
		
		\item Déterminer la matrice $K_{2}$ de la famille $\left(B_{2,0}, B_{2,1}, B_{2,2}\right)$ dans la base $\mathcal{C}_{2}$.
		\item En déduire que la famille $\left(B_{2,0}, B_{2,1}, B_{2,2}\right)$ est une base de $\mathbb{R}_{2}[X]$. 
		\item Calculer $T_{2}\left(A_{0}\right), T_{2}\left(A_{1}\right)$ et $T_{2}\left(A_{2}\right)$; déterminer la matrice $H_{2}$ de $T_{2}$ dans la base $\mathcal{C}_{2}$.
		\item Donner les racines du polynôme $P_{T_2}(X) = \text{det}(XI_3-H_2)$.
		\item Donner la dimension et une base pour chaque sous-espaces vectoriels  suivants: 
		
		$E_1$ = ker$\left(T_2-id\right)$  et  $E_{\frac{1}{2}}$ = ker$\left(T_2-\frac{1}{2}id\right)$
		
		(\textit{ce sont les sous-espaces propres associés aux valeurs propres trouvées à la question précédente})
		\textbf{Attention :} $E_1\subset \R[X]$ et $E_{\frac{1}{2}}\subset \R[X]$ 
	\end{enumerate}
	\item \textit{On revient au cas général où $n$ est un entier supérieur ou égal à 1.} 
	\begin{enumerate}
		\item Montrer que la famille $\left(B_{n, 0}, B_{n, 1}, \ldots, B_{n, n}\right)$ est libre ; en déduire que cette famille est une base de $\mathbb{R}_{n}[X]$.
		\item Montrer que l'application $T_{n}$ est un automorphisme de $\mathbb{R}_{n}[X]$.
		\item Calculer $T_{n}\left(A_{0}\right)$ et montrer que $T_{n}\left(A_{1}\right)=A_{1}$.
		\item Montrer que pour tout $k \in \llbracket 0, n \rrbracket$, le degré du polynôme $T_{n}\left(A_{k}\right)$ est égal à $k$.
		
		Pour établir ce résultat, on pourra utiliser la propriété suivante que l'on ne demande pas de démontrer:
		
		$$
		\forall k \in \llbracket 0, n-1 \rrbracket,\left(T_{n}\left(A_{k+1}\right)\right)(X)=\frac{1}{n} X(1-X)\left(T_{n}\left(A_{k}\right)\right)^{\prime}(X)+X\left(T_{n}\left(A_{k}\right)\right)(X)
		$$
		
		où $\left(T_{n}\left(A_{k}\right)\right)^{\prime}$ est le polynôme dérivé de $T_{n}\left(A_{k}\right)$.
		\item  Pour tout $k \in \llbracket 0, n \rrbracket$, soit $\alpha_{k}$ le coefficient de $X^{k}$ du polynôme $T_{n}\left(A_{k}\right)$. Calculer $\alpha_{k}$ en fonction de $k$ et $n$ et en déduire  $\text{det}(T_{n})$.
	\end{enumerate}
\end{enumerate}

\section{Partie II : Les polynômes d'interpolation de Lagrange}


\begin{enumerate}
	\setcounter{enumi}{2}
	\item Soit $n \in \mathbb{N}^{*}$ et $x_{0}, x_{1}, \ldots, x_{n}$ des réels deux à deux distincts. Soit $\Phi$ l'application de $\mathbb{R}_{n}[X]$ dans $\mathbb{R}^{n+1}$ telle que : $$\forall P \in \mathbb{R}_{n}[X], \Phi(P)=\left(P\left(x_{0}\right), P\left(x_{1}\right), \ldots, P\left(x_{n}\right)\right)$$
	 \begin{enumerate}
		\item Montrer que l'application $\Phi$ est un isomorphisme d'espaces vectoriels.
		\item On note $\left(e_{0}, e_{1}, \ldots, e_{n}\right)$ la base canonique de $\mathbb{R}^{n+1}$ avec $e_{0}=(1,0,0, \dots, 0),$ $ e_{1}=(0,1,0, \ldots, 0), \ldots \dots, e_{n}=(0,0,0, \dots, 1)$. Pour tout $i \in \llbracket 0, n \rrbracket$, on note $L_{i}$ le polynôme de $\mathbb{R}_{n}[X]$ tel que $\Phi\left(L_{i}\right)=e_{i}$.
		
		Pour tout $i \in \llbracket 0, n \rrbracket$, donner l'expression de $L_{i}$ en fonction des réels $x_{0}, x_{1}, \ldots, x_{n}$.
		
		\item Soit $\Psi$ l'application définie sur $\disp\left(\mathbb{R}_{n}[X]\right)^{2}$ par : $\disp \forall(P, Q) \in\left(\mathbb{R}_{n}[X]\right)^{2}, \Psi(P, Q)=\sum_{k=0}^{n} P\left(x_{k}\right) Q\left(x_{k}\right)$.
		
		Vérifier que $\Psi$ est un produit scalaire sur $\mathbb{R}_{n}[X]$. On munit alors $\mathbb{R}_{n}[X]$ de ce produit scalaire.
		
		\item Montrer que $\left(L_{0}, L_{1}, \ldots, L_{n}\right)$ est une base orthonormée de $\mathbb{R}_{n}[X]$.
		\item Expliciter la matrice $A$ de passage de la base $\left(L_{0}, L_{1}, \ldots, L_{n}\right)$ à la base canonique $\mathcal{C}_{n}$ de $\mathbb{R}_{n}[X]$. Donner son déterminant en fonction des réels $x_{0}, x_{1}, \ldots, x_{n}$.
		\item Soit $f$ une fonction continue sur $\mathbb{R}$ à valeurs réelles.
		
		Montrer que pour tout $n \in \mathbb{N}^{*}$, il existe un unique polynôme de $\mathbb{R}_{n}[X]$, noté $P_{f}$, vérifiant les relations :
		
		$$
		P_{f}\left(x_{0}\right)=f\left(x_{0}\right), P_{f}\left(x_{1}\right)=f\left(x_{1}\right), \ldots, P_{f}\left(x_{n}\right)=f\left(x_{n}\right)
		$$
		
		On dit que $P_{f}$ est le polynôme d'interpolation de la fonction $f$ aux points $x_{0}, x_{1}, \ldots, x_{n}$.
		
		Exprimer $P_{f}$ dans la base $\left(L_{0}, L_{1}, \ldots, L_{n}\right)$. 
	\end{enumerate}
	
	\item Soit $x_{0}, x_{1}, \ldots, x_{n}$ des réels appartenant à un intervalle $[a, b]$ tels que $a \leqslant x_{0}<x_{1}<\cdots<x_{n} \leqslant b$.
	
	Soit $f$ une fonction de classe $C^{n+1}$ sur $[a, b]$ et $\bar{x}$ un réel de $[a, b]$ différent de $x_{0}, x_{1}, \ldots, x_{n}$.
	
	On note $P_{f}$ le polynôme d'interpolation de la fonction $f$ aux points $x_{0}, x_{1}, \ldots, x_{n}$ et $Q_{f}$ le polynôme
	
	d'interpolation de la fonction $f$ aux points $x_{0}, x_{1}, \ldots, x_{n}, \bar{x}$. On pose $: \disp w(X)=\prod_{k=0}^{n}\left(X-x_{k}\right)$. 
	 \begin{enumerate}
		\item Établir l'existence d'un réel $\delta$ tel que pour tout $t \in[a, b]$, on a : $$Q_{f}(t)-P_{f}(t)=\delta \times w(t)$$
		(\textit{indication: quel est le degré du polynôme } $Q_{f}$ ?)
		
		\item Soit $h$ la fonction définie sur $[a, b]$ par : $\forall t \in[a, b], h(t)=f(t)-Q_{f}(t)$.
		
		Montrer que la fonction $h$ s'annule en les $(n+2)$ points $\bar{x}, x_{0}, x_{1}, \ldots, x_{n}$ et en déduire l'existence d'un réel $\theta \in] a, b\left[\right.$ tel que $h^{(n+1)}(\theta)=0$.
		\item Établir l'égalité : $$f(\bar{x})-P_{f}(\bar{x})=\frac{1}{(n+1) !} \times f^{(n+1)}(\theta) \times w(\bar{x})$$
		\item En déduire que pour tout $t \in[a, b]$, on a :
		 $$\left|f(t)-P_{f}(t)\right| \leqslant \frac{1}{(n+1) !} \times|w(t)| \times \sup _{[a, b]}\left|f^{(n+1)}\right|$$
	\end{enumerate}
	
\end{enumerate}
