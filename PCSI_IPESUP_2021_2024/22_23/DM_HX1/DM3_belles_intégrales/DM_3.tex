L'objectif de ce devoir est de vous amener à calculer trois "belles" limites:\\


\centering\rule{1\linewidth}{0.6pt}\\

\raggedright

\begin{enumerate}
	\item[$\square$] $\lim\limits_{n \rightarrow+\infty} \sum\limits_{k=1}^{n} \dfrac{1}{k^{2}}=\dfrac{\pi^{2}}{6}$ \hfill (zêta de Riemann en 2)
	\item[$\square$] $\lim\limits_{n \rightarrow+\infty} \dfrac{\sqrt{n}}{2^{2 n}}\begin{pmatrix}2 n \\ n\end{pmatrix}=\dfrac{1}{\sqrt{\pi}}$ \hfill(formule de Wallis)
	\item[$\square$] $\lim\limits_{n \rightarrow+\infty} \dfrac{1}{n !}\left(\dfrac{n}{e}\right)^{n} \sqrt{2 n \pi}=1$ \hfill
	(formule de Stirling)
\end{enumerate}
\centering\rule{1\linewidth}{0.6pt}\\

\section{Préliminaires}


Pour tout $n \in \mathbb{N}$, on pose: $a_{n}=\int_{0}^{\frac{\pi}{2}} \cos ^{n}(t) d t \quad$\hfill(intégrales de Wallis).

\begin{enumerate}

	\item Calculer $a_{0}$ et $a_{1}$, puis montrer que pour tout $n \in \mathbb{N}: a_{n}>0$.
	\item Montrer que pour tout $n \in \mathbb{N}: a_{n+2}=\dfrac{n+1}{n+2} a_{n}$.
\end{enumerate}


\section{Calcul de la valeur en 2 de la fonction zêta de Riemann}

Pour tout $n \in \mathbb{N}$, on pose : $b_{n}=\int_{0}^{\frac{\pi}{2}} t^{2} \cos ^{2 n}(t) d t$. 

\begin{enumerate}
	\item \begin{enumerate}
		\item Montrer que pour tout $t \in\left[0, \frac{\pi}{2}\right]: t \leqslant \frac{\pi}{2} \sin (t)$.
		\item En déduire que pour tout $n \in \mathbb{N}: 0 \leqslant b_{n} \leqslant \frac{\pi^{2}}{4}\left(a_{2 n}-a_{2 n+2}\right)$.
		\item En déduire enfin la limite: $\lim\limits_{n \rightarrow+\infty} \dfrac{b_{n}}{a_{2 n}}=0$.
	\end{enumerate}

\item \begin{enumerate}
	\item Montrer que pour tout $n \in \mathbb{N}: a_{2n+2}=(2n+2) \int_{0}^{\frac{\pi}{2}} t \sin (t) \cos ^{2 n+1}(t) d t$.
	\item  En déduire que pour tout $n \in \mathbb{N}: \dfrac{a_{2 n+2}}{n+1}=(2 n+1) b_{n}-(2 n+2) b_{n+1}$. 
	\item En déduire que pour tout $n \in \mathbb{N}: 2\left(\dfrac{b_{n}}{a_{2 n}}-\dfrac{b_{n+1}}{a_{2 n+2}}\right)=\dfrac{1}{(n+1)^{2}}$.
	\item En déduire enfin l'existence et la valeur de $\lim _{n \rightarrow+\infty} \sum\limits_{k=1}^{n} \dfrac{1}{k^{2}}$, que l'on note également: $\sum\limits_{k=1}^{+\infty} \dfrac{1}{k^{2}}$.
\end{enumerate}

\end{enumerate}

\section{Formule de Wallis}

Pour tout $n \in \mathbb{N}$, on pose : $\rho_{n}=\dfrac{a_{2 n}}{a_{2 n+1}}$.

\begin{enumerate}
	\item Montrer que pour tout $n \in \mathbb{N}: \rho_{n}=\dfrac{(2 n+1) \pi}{2^{4 n+1}}\begin{pmatrix}2 n \\ n\end{pmatrix}^{2}$.
	\item\begin{enumerate}
	\item Montrer que la suite $\left(a_{n}\right)_{n \in \mathbb{N}}$ est décroissante.
	\item En déduire un encadrement de $\rho_{n}$ pour tout $n \in \mathbb{N}$, puis la limite: $\lim\limits_{n \rightarrow+\infty} \rho_{n}=1$.
	\item En déduire la formule de Wallis.	
\end{enumerate}\end{enumerate}

\section{Formule de Stirling}


On note $f$ la fonction $x \longmapsto\left(x+\dfrac{1}{2}\right) \ln \left(1+\dfrac{1}{x}\right)$ et $g$ la fonction $x \longmapsto f(x)-\dfrac{1}{12 x}+\dfrac{1}{12(x+1)}$,

toutes deux définies sur $\mathbb{R}_{+}^{*}$. 


On pose également pour tout $n \in \mathbb{N}^{*}: u_{n}=\dfrac{n^{n+\frac{1}{2}}}{n ! e^{n}}$  \ et \  $v_{n}=\ln \left(u_{n}\right)$

\begin{enumerate}
	\item\begin{enumerate}
		\item Montrer que pour tout $x>0: f^{\prime \prime}(x)=\dfrac{1}{2 x^{2}(x+1)^{2}}$, et simplifier de même $g^{\prime \prime}$ sur $\mathbb{R}_{+}^{*}$.
		\item En déduire que $f$ est minorée par 1 , et $g$ majorée par 1 sur $\mathbb{R}_{+}^{*}$.
	\end{enumerate}
	\item\begin{enumerate}
		\item Exprimer $v_{n+1}-v_{n}$ à l'aide de la fonction $f$ pour tout $n \in \mathbb{N}^{*}$.
		\item En déduire que la suite $\left(v_{n}\right)_{n \in \mathbb{N}^{*}}$ est croissante et majorée, puis que la suite $\left(u_{n}\right)_{n \in \mathbb{N}^{*}}$ 
		
		converge vers un réel strictement positif $\ell$.
		\item Montrer, en étudiant pour tout $n \in \mathbb{N}^{*}$ le rapport $\dfrac{u_{n}^{2}}{u_{2 n}}$, que: $\ell=\dfrac{1}{\sqrt{2 \pi}}$. 
		
		En déduire la formule de Stirling.
		
	\end{enumerate}
\end{enumerate}
