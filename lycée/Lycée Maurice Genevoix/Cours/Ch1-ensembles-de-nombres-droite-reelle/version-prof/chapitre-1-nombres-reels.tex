\section{Les ensembles de nombres }

\subsection{Les nombres entiers : entiers naturels et entiers relatifs}

\medskip 

\noindent\fcolorbox{vert}{white}{
	\begin{minipage}{1\linewidth}
		\begin{defi}
			Un entier naturel est un nombre entier positif ou nul.\\
			L'ensemble des entiers naturels est noté $\mathbb{N}$ (\og N \fg comme \og naturel \fg).
		\end{defi}
	\end{minipage}
}

\begin{rmq}
	$$\mathbb{N}=\{0, 1, 2, 3, ...\}$$
	En particulier, $0 \in\mathbb{N}$. 
\end{rmq}

\medskip

\noindent\fcolorbox{vert}{white}{
	\begin{minipage}{1\linewidth}
		\begin{defi}
			Un entier relatif est un nombre entier positif ou négatif ou nul.\\
			L'ensemble des entiers relatifs est noté $\mathbb{Z}$ (\og Z \fg comme \og zahlen \fg qui signifie \og compter \fg en allemand).
		\end{defi}
	\end{minipage}
}

\begin{rmq}
	$$\mathbb{Z}=\{..., -3 , -2 , -1 , 0, 1, 2, 3, ...\}$$
\end{rmq}
\hfill\\[-0.2cm]

\noindent\fcolorbox{rouge}{white}{
	\begin{minipage}{1\linewidth}
		\begin{prop}
			Tout entier naturel est un entier relatif. On dit que l'ensemble des entiers naturels \textbf{est inclus} dans l'ensemble des entiers relatifs et on note $\mathbb{N}\subset\mathbb{Z}.$\\
			$$\begin{tikzpicture}[scale=1.5]
			\draw[red](0,0) ellipse [x radius=1cm, y radius=0.3cm];
			\draw[red](0.5,0.05) node{$\mathbb{N}$};
			\draw[blue](0,0) ellipse [x radius=1.5cm, y radius=0.5cm];
			\draw[blue](1.2,0.05) node{$\mathbb{Z}$};
			\end{tikzpicture}$$
		\end{prop}
	\end{minipage}
}

\begin{proof}\hfill\\
	Par définition, tout entier naturel est un entier positif ou nul. Donc, en particulier, c'est un entier positif ou négatif ou nul.
\end{proof}

\begin{rmq}
	L'inclusion réciproque est fausse : $\mathbb{Z} \not\subset \mathbb{N}$. Pour le prouver, il suffit de trouver un élément de $\mathbb{Z}$ qui n'est pas un élément de $\mathbb{N}$. Le nombre $-5$ convient (par exemple).\\
\end{rmq}

\noindent\textcolor{red}{\textbf{Attention à ne pas confondre les deux symboles $\in$ et $\subset$.}}\\

\begin{exer}
	Dans chacun des cas suivants, compléter les \ldots à l'aide du symbole $\in$ ou du symbole $\subset$.
	\begin{enumerate}[$\diamond$]
		\item $12$ \ldots $\mathbb{N}$
		\item soit $(d)$ une droite et $M$ un point de cette droite. Alors $M$ \ldots $(d)$.  
		\item soient $A$ et $B$ deux points distincts du plan. Alors $[AB] \ldots (AB)$.
		\item soit $E$ l'ensemble de tous les élèves du lycée et $S$ l'ensemble des élèves de votre classe de seconde. Alors $S \ldots E$
		\item soit $P$ l'ensemble de tous les professeurs du lycée. Alors $\text{Mme Beudez}\ldots P$ 
	\end{enumerate}
\end{exer}


\begin{rmq}
	.\\
	L'inclusion est une relation entre deux ensembles. Il ne faut pas la confondre avec l'appartenance (qui est une relation entre un élément et un ensemble). Par exemple, comme $3$ est un nombre et pas un ensemble de nombres, on dit que \og $3$ appartient à $\mathbb{N}$ \fg, ce qui se note \og $3 \in \mathbb{N}$ \fg.\\
\end{rmq}

\subsection{Les nombres décimaux}

\medskip 

\noindent\fcolorbox{vert}{white}{
	\begin{minipage}{1\linewidth}
		\begin{defi}
			Un nombre \textbf{décimal} est un nombre pouvant s'écrire sous la forme $\di\frac{a}{10^{n}}$ où $a$ est un entier \textbf{relatif} et $n$ est un entier \textbf{naturel}.\\
			L'ensemble des nombres décimaux est noté $\mathbb{D}$ (\og D \fg comme \og décimal \fg).\\
			Autrement dit, $\mathbb{D}=\left\{\dfrac{a}{10^{n}} \ ; \ a\in\mathbb{Z} \ \text{et} \ n\in\mathbb{N}\right\}$.
		\end{defi}
	\end{minipage}
}
\hfill\\


\begin{ex} \hfill
	\begin{enumerate}[$\diamond$]
		\item $0,2$ est un nombre décimal (donc $0,2\in\mathbb{D}$). En effet, $0,2=\di\frac{2}{10^{1}}$ avec $2$ entier relatif et $1$ entier naturel.
		\item $-0,04$ est un nombre décimal (donc $-0,04\in\mathbb{D}$). En effet, $-0,04=\di\frac{-4}{10^{2}}$ avec $-4$ entier relatif et $2$ entier naturel.\\
	\end{enumerate}
\end{ex}
\hfill\\


\noindent\fcolorbox{rouge}{white}{
	\begin{minipage}{1\linewidth}
		\begin{prop}
			Tout entier relatif est un nombre décimal. On a donc les inclusions suivantes : $\mathbb{N}\subset\mathbb{Z}\subset\mathbb{D}$
			$$\begin{tikzpicture}[scale=1.5]
			\draw[red](0,0) ellipse [x radius=1cm, y radius=0.3cm];
			\draw[red](0.5,0.05) node{$\mathbb{N}$};
			\draw[blue](0,0) ellipse [x radius=1.5cm, y radius=0.5cm];
			\draw[blue](1.2,0.05) node{$\mathbb{Z}$};
			\draw[vert](0,0) ellipse [x radius=2cm, y radius=0.7cm];
			\draw[vert](1.75,0.05) node{$\mathbb{D}$};
			\end{tikzpicture}$$
		\end{prop}
	\end{minipage}
}

\begin{proof}
	Pour tout entier relatif $k$, on a :
	$$k = \di\frac{k}{1}=\di\frac{k}{10^{0}}$$
	Ainsi, tout entier relatif $k$ peut s'écrire sous la forme $\di\frac{k}{10^{0}}$ avec $k$ entier relatif et $0$ entier naturel. Tout entier relatif est donc un nombre décimal, ce qui prouve bien que $\mathbb{Z} \subset \mathbb{D}$. L'inclusion $\mathbb{N} \subset \mathbb{Z}$ ayant déjà été vue précédemment, on obtient bien $\mathbb{N}\subset\mathbb{Z}\subset\mathbb{D}$. \\
\end{proof}

\begin{rmq} 
	Les nombres décimaux ont une écriture décimale finie (c'est-à-dire qu'ils s'écrivent avec un nombre fini de chiffres après la virgule). Réciproquement, un nombre ayant une écriture décimale finie est un nombre décimal.
\end{rmq}

\noindent\fcolorbox{rouge}{white}{
	\begin{minipage}{1\linewidth}
\begin{prop}
	Le nombre $\di\frac{1}{3}$ n'est pas décimal.
\end{prop}
\end{minipage}
}

\begin{proof}
	%Admise pour l'instant. La démonstration sera faite plus tard lorsque nous aurons vu la notion de diviseur et de nombre premier.
	Avant de commencer la démonstration, on rappelle le critère de divisibilité par $3$ : 
	un nombre est divisible par $3$ si et seulement si la somme de ses chiffres est divisible par $3$.\\
	On raisonne \textbf{par l'absurde} : supposons que  $\di\frac{1}{3}$ soit décimal. Alors il existe un entier relatif $a$ et un entier naturel $n$ tels que $\di\frac{1}{3}=\dfrac{a}{10^{n}}$. D'où $3 \times a = 10^{n}$. En particulier, $a$ étant un entier, on en déduit que $3$ divise $3 \times a$. Or $3 \times a =10^{n}$ donc \\[0.15cm]
	$3$ divise $10^{n}$. Or la somme des chiffres de $10^{n}$ vaut $1$, qui n'est pas un multiple de $3$. Cela contredit le critère de divisibilité par $3$. On a donc démontré que $\dfrac{1}{3}$ n'est pas décimal.\\
\end{proof}

\subsection{L'ensemble des nombres rationnels}

\medskip

\noindent\fcolorbox{vert}{white}{
	\begin{minipage}{1\linewidth}
		\begin{defi} \hfill
			\begin{enumerate}[$\diamond$]
				\item Un nombre \textbf{rationnel} est un nombre qui peut s'écrire sous la forme $\di\frac{p}{q}$ où $p$ est un entier relatif et $q$ un entier naturel non nul.\\
				L'ensemble des nombres rationnels est noté $\mathbb{Q}$ ( \og Q \fg comme \og quotient \fg).\\
				Autrement dit, $\mathbb{Q}=\left\{\dfrac{p}{q} \ ; \ p\in\mathbb{Z} \ \text{et} \ q\in\mathbb{N}^{*}\right\}$.
				\item Un nombre qui n'est pas rationnel est dit \textbf{irrationnel}.
			\end{enumerate}
		\end{defi}
	\end{minipage}
}
\hfill\\

\noindent\fcolorbox{rouge}{white}{
	\begin{minipage}{1\linewidth}
\begin{prop}
	On a les inclusions suivantes :
	$$\mathbb{N}\subset\mathbb{Z}\subset\mathbb{D}\subset\mathbb{Q}.$$
	$$\begin{tikzpicture}[scale=1.5]
	\draw[red](0,0) ellipse [x radius=1cm, y radius=0.3cm];
	\draw[red](0.5,0.05) node{$\mathbb{N}$};
	\draw[blue](0,0) ellipse [x radius=1.5cm, y radius=0.5cm];
	\draw[blue](1.2,0.05) node{$\mathbb{Z}$};
	\draw[vert](0,0) ellipse [x radius=2cm, y radius=0.7cm];
	\draw[vert](1.75,0.05) node{$\mathbb{D}$};
	\draw[purple](0,0) ellipse [x radius=2.5cm, y radius=0.9cm];
	\draw[purple](2.15,0.05) node{$\mathbb{Q}$};	
	\end{tikzpicture}$$
\end{prop}
	\end{minipage}
}
\hfill\\[-0.5cm]

\begin{proof}
	Tout nombre décimal $d$ peut s'écrire sous la forme $\dfrac{a}{10^n}$ où $a$ est un entier relatif et $n$ un entier naturel. On pose alors $p=a$ et $q=10^n$. Ainsi, $p$ est un entier relatif et $q$ est un entier naturel non nul. Donc $d$ est un nombre rationnel.\\
	Tout nombre décimal est donc un nombre rationnel, ce qui prouve bien que $\mathbb{D} \subset \mathbb{Q}$. Les inclusions $\mathbb{N} \subset \mathbb{Z} \subset \mathbb{D}$ ayant déjà été vues précédemment, on obtient bien $\mathbb{N}\subset\mathbb{Z}\subset\mathbb{D} \subset\mathbb{Q}$. \\
\end{proof}
\hfill\\

\noindent\fcolorbox{rouge}{white}{
	\begin{minipage}{1\linewidth}
		\begin{prop}
			Tout nombre rationnel s'écrit de manière unique sous forme irréductible. Autrement dit, pour tout nombre rationnel $r$, il existe un unique entier relatif $a$ et un unique entier naturel non nul $b$ tel que $r=\dfrac{a}{b}$ et tels que le seul diviseur positif commun à $a$ et $b$ soit $1$.
		\end{prop}
	\end{minipage}
}


\begin{proof}
	Soit $r$ un nombre rationnel. Il existe $p\in\mathbb{Z}$ et $q\in\mathbb{N}^{*}$ tel que $r=\dfrac{p}{q}$. On considère $d$ le plus grand diviseur commun à $p$ et à $q$. Alors, par définition, $d$ divise $p$ et $d$ divise $q$ donc il existe un entier relatif $a$ et un entier naturel non nul $b$ tels que :
	$$r = \dfrac{p}{q}=\dfrac{d \times a}{d \times b} = \dfrac{a}{b}$$
	La fraction $\dfrac{a}{b}$ est nécessairement irréductible puisque $d$ était le \textbf{plus grand divisieur commun} de $p$ et de $q$. 
\end{proof}
\hfill\\

\subsection{L'ensemble des réels, la droite réelle}

\medskip

\noindent\fcolorbox{blue}{white}{
	\begin{minipage}{1\linewidth}
\noindent \underline{Point historique :} \\
La mesure des grandeurs est à l'origine de l'invention des nombres réels. Ils permettent d'associer à toute mesure une valeur. Une unité de longueur étant choisie :
\begin{enumerate}[$\bullet$]
	\item toute longueur de segment est mesurée par un réel nombre positif. Par exemple la longueur de la diagonale d'un carré de côté $1$ est mesurée par le réel positif $\sqrt{2}$.
	\item inversement, tout réel positif est la mesure d'une longueur de segment.
\end{enumerate} 
	\end{minipage}
}
\hfill\\
\\

\subsection{Encadrement décimal d'un nombre réel à $10^{-n}$ près}

\medskip

\noindent\fcolorbox{rouge}{white}{
	\begin{minipage}{1\linewidth}
		\begin{prop-def}
			Soit $x$ un nombre réel et $n$ un entier relatif. Il existe un unique nombre entier relatif $a$ tel que :
			$$\dfrac{a}{10^{n}} \leqslant x < \dfrac{a+1}{10^n}$$
			Cet encadrement est appelé l'\textbf{encadrement décimal de $x$ à $10^{-n}$ près}.\\[0.1cm]
			\textbf{L'arrondi de $x$ à $10^{-n}$ près} est celui des deux nombres $\dfrac{a}{10^n}$ ou $\dfrac{a+1}{10^n}$ qui est le plus proche de $x$ (lorsqu'il existe).\\[0.1cm]
			Dans le cas où $x$ est à égale distance de $\dfrac{a}{10^n}$ et de $\dfrac{a+1}{10^n}$, par convention, l'arrondi est alors $\dfrac{a+1}{10^n}$.
		\end{prop-def}
	\end{minipage}
}
\hfill\\

\begin{proof}
	Admis.
\end{proof}

\subsection{Quelques exemples de constructions géométriques de nombres réels}

\medskip 

\noindent Il est très facile de placer un entier sur la droite des réels : il suffit de reporter l'unité de longueur autant de fois que nécessaire sur la droite, en faisant bien attention à l'orientation de la droite.\\

\subsection{Intersection et réunion d'intervalles}

\medskip



\noindent\fcolorbox{vert}{white}{
	\begin{minipage}{1\linewidth}
		\begin{defi}Soient $I$ et $J$ deux intervalles de $\mathbb{R}$.
			\begin{enumerate}[$\diamond$]
				\item L'\textbf{intersection} de $I$ et de $J$ est l'ensemble des nombres qui appartiennent à la fois à $I$ et à $J$. On la note $I\cap J$, ce qui se lit "$I$ inter $J$".
				\item La \textbf{réunion} de $I$ et de $J$ est l'ensemble des nombres qui appartiennent à $I$ ou à $J$ (éventuellement aux deux à la fois). On la note $I\cap J$, ce qui se lit "$I$ union $J$".
			\end{enumerate}
		\end{defi}
	\end{minipage}
}
\hfill\\

\begin{rmq}
	Lorsque les intervalles $I$ et $J$ n'ont aucun nombre en commun, leur intersection est l'ensemble vide, noté $\emptyset$. On dit alors que les intervalles $I$ et $J$ sont \textbf{disjoints}.\\
\end{rmq}

\begin{rmq}
	On en déduit immédiatement que $I\cap J\subset I\cup J$.\\
\end{rmq}

\noindent\fcolorbox{blue}{white}{
	\begin{minipage}{1\linewidth}
		\noindent\underline{\textbf{Méthode pour représenter l'intersection et l'union de deux intervalles :}}\\[0.1cm]
		On commence par représenter ces deux intervalles sur la même droite avec deux couleurs différentes. 
		\begin{enumerate}[$\diamond$]
			\item L'intersection des deux intervalles est la partie de la droite qui a été coloriée \textbf{des deux couleurs à la fois}.
			\item La réunion des deux intervalles est la partie de la droite qui a été coloriée \textbf{d'au moins une des deux couleurs} (soit d'une couleur uniquement, soit des deux couleurs à la fois).
		\end{enumerate}
	\end{minipage}
}
\hfill\\


\begin{exer}\hfill\\[0.1cm]
	\parbox{0.5\linewidth}{
		\begin{enumerate}[$\diamond$]
			\item $I=\left[\vtab3;7\right]$ et $J=\left[\vtab4;9\right[$. Alors $I\cap J=\ldots \ldots$.\\[-0.1cm]
			\item $I=\left[\vtab -5;2\right[$ et $J=\left]\vtab3;4\right[$. Alors $I\cap J=\ldots \ldots$.\\[-0.1cm]
			\item $I=\left[\vtab 2;6\right]$ et $J=\left]\vtab3;+\infty\right[$. Alors  $I\cap J=\ldots \ldots$.\\[-0.1cm]
			\item $I=]-2;3]$ et $J=[3;7]$. Alors  $I\cap J=\ldots \ldots$.\\[-0.1cm]
		\end{enumerate}}
	\parbox{0.5\linewidth}{
		\begin{enumerate}[$\diamond$]
			\item $I=[3;7]$ et $J=[4;9[$. Alors $I\cup J=\ldots \ldots$\\[-0.1cm]
			\item $I=[1;3]$ et $J=[-2;10]$. Alors $I\cup J=\ldots \ldots$\\[-0.1cm]
			\item $I=[-5;2[$ et $J=]3;4[$. Alors $I\cup J=\ldots \ldots$\\[-0.1cm]
			\item $I=[0;4]$ et $J=]-\infty;1]$. Alors $I\cup J=\ldots \ldots$\\[-0.1cm]

\section{Valeur absolue d'un nombre réel. Distance entre deux nombres réels.}

\medskip

%\subsection{Valeur absolue d'un nombre réel}

%\medskip

\noindent\fcolorbox{vert}{white}{
	\begin{minipage}{1\linewidth}
		\begin{defi}
			Soit $x$ un nombre réel.\\
			On appelle \textbf{valeur absolue} de $x$, et on note $|x|$, le nombre réel égal à : $\begin{cases} x \ \ \text{si $x \geqslant 0$} \\ -x \ \ \text{si $x<0$}\end{cases}$.
		\end{defi}
	\end{minipage}
}
\hfill\\[-0.2cm]

\begin{ex} \hfill\\
	$|6|=6$ \ \ ; \ \ $|-5|=5$ \ \ ; \ \ $|-3,2|=3,2$ \ \ ; \ \ $|0|=0$
\end{ex}
\hfill\\[-0.4cm]

\noindent\fcolorbox{vert}{white}{
	\begin{minipage}{1\linewidth}
		\begin{defi}
			Soient $a$ et $b$ deux nombres réels.\\
			On appelle \textbf{distance entre les réels $a$ et $b$} le nombre $|b-a|$ (aussi égal à $|a-b|$). 
		\end{defi}
	\end{minipage}
}
\hfill\\

\begin{ex}
	La distance entre les nombres $1$ et $6$ est $|6-1|=|5|=5$.\\
	La distance entre les nombres $5$ et $0,3$ est $|0,3-5|=|-4,7|=4,7$.\\
	La distance entre les nombres $-2$ et $4$ est $|4-(-2)|=|4+2|=|6|=6$.\\
\end{ex}

\begin{exer}
	On considère les réels suivants :
	$$a=-4 \ \ ; \ \ b=-1 \ \ ; \ \ c=2 \ \ ; \ \ d=3,6$$
	\begin{enumerate}
		\item Calculer : 
			\begin{enumerate}[$\diamond$] 
				\item la distance entre $a$ et $b$ ;
				\item la distance entre $c$ et $d$ ;
				\item la distance entre $a$ et $c$.
			\end{enumerate}
		\item Sur une droite de repère $(O;I)$, placer les points $A$, $B$, $C$ et $D$ d'abscisses respectives $a$, $b$, $c$ et $d$.\\
		Que semblent représenter géométriquement les résultats obtenus à la question précédente ?\\
	\end{enumerate}
\end{exer}

\noindent\underline{\textbf{Interprétation géométrique de $|b-a|$ avec $a$ et $b$ deux réels :}}\\[0.1cm]
On considère une droite graduée $(d)$. On note $A$ le point d'abscisse $a$ et $B$ le point d'abscisse $b$. Alors la distance $|b-a|$ entre les nombres $a$ et $b$ s'interprète comme la distance $AB$.\\[0.1cm]


\noindent\fcolorbox{rouge}{white}{
	\begin{minipage}{1\linewidth}
		\begin{prop}
			Soient $a$, $x$ et $r$ trois nombres réels avec $r\geqslant 0$. Alors : 
			$$x\in[a-r ; a+r] \ \ \text{équivaut à} \ \ |x-a|\leqslant r$$
		\end{prop}
	\end{minipage}
}
\hfill\\

\begin{proof}
	$$\begin{tikzpicture}[scale=2.5]
	\draw [->] (0,0) -- (5,0) ;
	\draw [red, line width= 1.5 pt] (1,0) -- (4,0) ;

	\draw[red] (4,0) node{$]$} node[below=3pt]{$a+r$};
	\draw[red] (2.5,0) node{$+$} node[below=3pt]{$a$};
	\draw (2,0) node{$\times$} node[below=3pt]{$x$};

	\draw[->] (2.5,0.2)--(1,0.2);
	\draw[->] (2.5,0.2)--(4,0.2);
	\draw[->] (4,0.2)--(2.5,0.2);
	\draw (1.75,0.2) node[above=3pt]{$r$};
	\draw (3.25,0.2) node[above=3pt]{$r$};

	%\draw [decorate,decoration={brace,amplitude=5pt,mirror,raise=4ex}]

\end{proof}
