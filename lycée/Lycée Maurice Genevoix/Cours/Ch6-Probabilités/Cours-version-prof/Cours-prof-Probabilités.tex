	\end{defi}\end{minipage}\hfill\\}

			\end{itemize}
		\end{prop}
	\end{minipage}

\noindent\fcolorbox{vert}{white}{\begin{minipage}{1\linewidth}\begin{defi}\hfill\\[0.15cm]  
Soient $A$ et $B$ deux évènements d'une expérience aléatoire.\\[0.25cm]
\begin{minipage}[t]{0.48\linewidth}
	$\square$ \quad L'évènement $A\cap B$ est appelé .\dotfill\\[0.15cm]
	.\dotfill\\[0.15cm]

		$\square$ \quad L'évènement $A\cup B$ est appelé .\dotfill\\[0.15cm]
		.\dotfill\\[0.15cm]
		Il est constitué des issues réalisant uniquement \\[0.15cm] l'évènement $A$ \textbf{ou} uniquement l'évènement $B$ \textbf{ou}\\[0.15cm] les deux.\\

\noindent\fcolorbox{vert}{white}{\begin{minipage}{1\linewidth}\begin{defi}\hfill\\[0.25cm]  
			Un \textbf{arbre} permet de représenter et dénombrer les issues d'une expérience aléatoire, en particulier lorsqu'il y a une succession de plusieurs épreuves. \\[0.1cm]     
	\end{defi}\end{minipage}\hfill\\}

		\item Donner le nombre d'issues possible pour cette expérience aléatoire.\\[0.3cm]
		.\dotfill \\[0.1cm]

		.\dotfill \\[0.1cm]
