\section{Proportions et pourcentages}


\medskip 


\subsection{Proportion d'un sous-ensemble}
 
\medskip

\noindent\fcolorbox{vert}{white}{
	\begin{minipage}{1\linewidth}
		\begin{defi}\hfill\\
			Soit $E$ un ensemble de référence non vide (appelé \textbf{population}) et $N$ le nombre d'éléments de $E$ (appelé l'effectif de $E$).\\
			Soit $A$ une partie de $E$ (appelée \textbf{sous-population}) et $n$ le nombre d'éléments de $A$ (appelé l'effectif de $A$).\\
			On appelle \textbf{proportion de $A$ dans $E$} le nombre réel $p$ défini par :\\
			\parbox{0.5\linewidth}{$$\fbox{\text{$p=\dfrac{n}{N}$}}$$}
			\parbox{0.5\linewidth}{ 
				$$\begin{tikzpicture}[scale=1.5]
				\draw[red](0,0) ellipse [x radius=1cm, y radius=0.3cm];
				\draw[red](0.5,0.05) node{$A$};
				\draw[blue](0,0) ellipse [x radius=2cm, y radius=0.75cm];
				\draw[blue](1.75,0.05) node{$E$};
				\end{tikzpicture}$$}
		\end{defi}
	\end{minipage}
}
\hfill\\

\begin{rmq}\hfill
	\begin{enumerate}[$\diamond$]
		\item Cette quantité $p$ est aussi appelée la fréquence de $A$ dans $E$.
		\item Si l'on connait deux des trois quantités parmi $p$, $n$ et $N$, on peut facilement calculer la $3^{\text{ème}}$ à l'aide d'un produit en croix :\\
		\renewcommand{\arraystretch}{1}
		$$\begin{tabular}{|c|c|c|}
		\hline
		expression de $p$ en & expression de $n$ en & expression de $N$ en \\
		fonction de $n$ et de $N$ & fonction de $p$ et de $N$ & fonction de $p$ et de $n$ \\
		\hline
		& & \\
		$p=\dfrac{n}{N}$ & $n = p \times N$ & $N=\dfrac{n}{p}$ \\
		& & \\
		\hline
		\end{tabular}$$
	\end{enumerate}
\end{rmq}
\hfill\\

\begin{ex}
	\begin{enumerate}[$\diamond$]
		\item Dans une classe de $32$ élèves, $8$ élèves jouent de la musique.\\
		La proportion d'élèves de la classe jouant de la musique est : $\dfrac{8}{32}=\dfrac{1}{4}$.
		\item Dans un lycée, la proportion $p$ d'externes est $0,19$. On sait que cela représente $285$ élèves. On cherche le nombre total d'élèves dans ce lycée.\\ % LLS exo 23 p.259
		D'après l'énoncé, $p=0,19$ et $n=285$ et on cherche $N$. On a $N=\dfrac{n}{p}=\dfrac{285}{0,19}=\np{1500}$
	\end{enumerate}
\end{ex}

\noindent\fcolorbox{rouge}{white}{
	\begin{minipage}{1\linewidth}
		\begin{prop}\hfill\\
			Soit $E$ un ensemble non vide et $A$ une partie de $E$. Alors la proportion $p$ de $A$ dans $E$ vérifie :
			$$0 \leqslant p \leqslant 1$$
			\hfill\\[0.2cm]
		\end{prop}
	\end{minipage}
}
\hfill\\

\begin{proof}\hfill
	\begin{enumerate}[$\diamond$]
		\item On commence par démontrer que $p\geqslant 0$.\\
		Avec les notations de la définition précédente, $p=\dfrac{n}{N}$ avec $n$ l'effectif de $A$ et $N$ l'effectif de $E$. Ainsi, les nombres $n$ et $N$ sont des entiers naturels, avec $N\not=0$ puisque $E$ n'est pas vide. Donc $p$ est positif comme le quotient d'un nombre positif ou nul par un nombre strictement positif.
		\item Il reste à démontrer que $p\leqslant 1$. Comme $A$ est une partie de $E$, on a $n \leqslant N$ (avec égalité si $A=E$). Donc, en divisant les deux membres de cette inégalité par le nombre strictement positif $n_E$, on obtient $\dfrac{n}{N}\leqslant 1$, d'où $p\leqslant 1$.
	\end{enumerate}
\end{proof}
\hfill\\


%\subsection{Expression d'une proportion en pourcentage}

\begin{rmq}
	On exprime souvent $p$ en pourcentage pour que le résultat soit plus parlant.\\
	Pour exprimer le nombre $p$ en pourcentage, il suffit de le multiplier par $100$.  Ainsi, dans l'exemple précédent, le pourcentage d'élèves de la classe jouant de la musique est $25\%$ puisque $p \times 100 = \dfrac{1}{4} \times 100 =0,25 \times 100 = 25$.\\
	Inversement, si une proportion est exprimée en pourcentage, on peut l'exprimer sous forme décimale en la divisant par $100$ :
	\begin{enumerate}[$\bullet$]
		\item Une proportion de $65,3 \%$ correspond à $p=0,653$.
		\item Une proportion de $20\%$ correspond à $p=0,2$.
	\end{enumerate}
\textcolor{red}{Attention, lorsque vous utilisez une formule utilisant une proportion $p$, cette proportion ne doit surtout pas être exprimée en $\%$ au moment d'appliquer la formule.}
\end{rmq}
\hfill\\

\noindent\fcolorbox{rouge}{white}{
	\begin{minipage}{1\linewidth}
		\begin{prop} (pourcentage d'une quantité)\hfill\\
			Si une quantité $q$ représente $t\%$ d'une quantité $Q$, alors $q=\dfrac{t}{100} \times Q$.
		\end{prop}
	\end{minipage}
}
\hfill\\

\begin{proof}
	Cela découle immédiatement de la définition de proportion.\\
	La proportion $p$ de $q$ dans $Q$ est $p=\dfrac{q}{Q}$ donc $q=p \times Q$. Ici, $p=\dfrac{t}{100}$ donc $q=\dfrac{t}{100} \times Q$. 
\end{proof}
\hfill\\

\begin{rmq}
	Autrement dit, de manière imagée, on peut voir le symbole  $"\%"$  comme un synonyme de  $\times \dfrac{1}{100}$. 
\end{rmq}

\begin{ex}
	\og $20\%$ de $45$ \fg se traduit mathématiquement par \og $20 \times \dfrac{1}{100} \times 45$ \fg, ce qui est égal à $9$.
\end{ex}
\hfill\\[-0.2cm]

\noindent\fcolorbox{rouge}{white}{
	\begin{minipage}{1\linewidth}
		\begin{prop} \hfill\\
			Soit $E$ un ensemble non vide, $A$ une partie de $E$ et $B$ une partie de $A$.\\
			On note $p_A$ la proportion de $A$ dans $E$ et $p_B$ la proportion de $B$ dans $A$.\\

		\end{prop}

	Attention, les proportions ne doivent pas être exprimées en pourcentage au moment du calcul.
\end{rmq}

\subsubsection{Variation absolue}

\medskip

\noindent\fcolorbox{vert}{white}{
	\begin{minipage}{1\linewidth}
		\begin{defi}\hfill\\
			On considère une quantité qui varie au cours du temps. On note :
			\begin{enumerate}[$\diamond$]
				\item $V_{d}$ la valeur de départ (on dit aussi valeur initiale) de la quantité ;

		\end{defi}
	\end{minipage}

\subsection{Taux d'évolution (ou variation relative)}

\medskip

\noindent\fcolorbox{vert}{white}{
	\begin{minipage}{1\linewidth}
		\begin{defi}\hfill\\
			On considère une quantité qui varie au cours du temps. On note :
			\begin{enumerate}[$\diamond$]
				\item $V_{d}$ la valeur de départ (on dit aussi valeur initiale) de la quantité ;
				\item $V_{a}$ la valeur d'arrivée (on dit aussi valeur finale) de la quantité .
			\end{enumerate}
			On appelle \textbf{taux d'évolution} (ou encore \textbf{variation relative}) de cette quantité le nombre :
			$$\di\frac{V_{a} - V_{d}}{V_d}$$
		\end{defi}
	\end{minipage}
}
\\[0.4cm]

\begin{rmq}
	La variation relative (ou taux d'évolution) n'a pas d'unité, mais on peut l'exprimer en pourcentage en multipliant sa valeur par $100$.
\end{rmq}
\hfill\\

\noindent\fcolorbox{rouge}{white}{
	\begin{minipage}{1\linewidth}
		\begin{prop}\hfill
			\begin{enumerate}[$\diamond$]
				\item Lorsque le taux d'évolution d'une quantité est positif, la quantité augmente.
				\item Lorsque le taux d'évolution d'une quantité est négatif, la quantité diminue.
			\end{enumerate}
		\end{prop}
	\end{minipage}
}
\\[0.4cm]

\medskip

\noindent Le coefficient multiplicateur est un outil très efficace pour la résolution de problèmes faisant étudier des évolutions. C'est lui qui permet de faire tous les calculs d'évolution.\\
\\
