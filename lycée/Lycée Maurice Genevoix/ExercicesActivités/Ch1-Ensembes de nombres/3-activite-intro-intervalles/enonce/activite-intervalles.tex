\noindent Dans chacune des questions suivantes, on considère une droite $(d)$ graduée (c'est-à-dire munie d'un repère).\\

\begin{enumerate}
	\item Repasser en couleur sur cette droite l'ensemble des points $M$ dont l'abscisse vérifie $-2 \leqslant x \leqslant 3$.\\
	
	$$\begin{tikzpicture}[scale=0.75]
	\draw [-> , line width = 1pt] (-8,0) -- (8,0) ;
	\draw (0,0) node[below=3pt]{$0$};
	\draw (1,0) node[below=3pt]{$1$};
	\foreach \x in {-7,...,7}
	{
	%	\draw (\x,0) node[below=3pt]{$\x$};
		\draw [line width = 1pt] (\x,-0.1) -- (\x,0.1) ;
	}
	\draw (-8,0) node[left=3pt]{$-\infty$};
	\draw (8,0) node[right=3pt]{$+\infty$};
	\end{tikzpicture}$$
	
	Le nombre $-2$ doit-il être colorié ? Pourquoi ? \\
	\\
	.\dotfill \\
	\\
	Le nombre $3$ doit-il être colorié ? Pourquoi ? \\
	\\
	.\dotfill \\
	\\
	Cette \og zone coloriée \fg est appelée un \textbf{intervalle}. On le note \ldots \ldots \ldots \ldots. \\[0.2cm]
	
	\begin{rmq}
		Pour écrire un intervalle, on utilise toujours des \ldots \ldots \ldots \ldots.\\[0.3cm]
		Ici, ils sont tous les deux \ldots \ldots \ldots \ldots \ldots (ils \og \ldots \ldots \ldots \ldots \ldots \ldots \ldots \ldots \ldots \fg la zone coloriée), ce qui indique \\[0.3cm]
		que $-2$ et $3$ \ldots \ldots \ldots \ldots \ldots \ldots \ldots \ldots à l'intervalle.
	\end{rmq}
	\hfill\\
	
	\item Repasser en couleur sur cette droite l'ensemble des points $M$ dont l'abscisse vérifie $-5 < x \leqslant -3$.\\
	
	$$\begin{tikzpicture}[scale=0.75]
	\draw [-> , line width = 1pt] (-8,0) -- (8,0) ;
	\draw (0,0) node[below=3pt]{$0$};
	\draw (1,0) node[below=3pt]{$1$};
	\foreach \x in {-7,...,7}
	{
		%	\draw (\x,0) node[below=3pt]{$\x$};
		\draw [line width = 1pt] (\x,-0.1) -- (\x,0.1) ;
	}
	\draw (-8,0) node[left=3pt]{$-\infty$};
	\draw (8,0) node[right=3pt]{$+\infty$};
	\end{tikzpicture}$$
	
	Le nombre $-5$ doit-il être colorié ? Pourquoi ? \\
	\\
	.\dotfill \\
	\\
	Le nombre $-3$ doit-il être colorié ? Pourquoi ? \\
	\\
	.\dotfill \\
	\\
	On note cet intervalle \ldots \ldots \ldots \ldots \\[0.2cm]
	Ici, le \ldots \ldots \ldots de droite est \ldots \ldots \ldots \ldots \ldots tandis que celui de gauche est \ldots \ldots \ldots \ldots \ldots (il \og \ldots \ldots \ldots \ldots \\[0.3cm]
	\ldots \ldots \ldots \ldots \ldots \ldots \ldots \ldots \ldots \fg la zone coloriée) \\
	\\
	
	
	\item Repasser en couleur sur cette droite l'ensemble des points $M$ dont l'abscisse vérifie $x \geqslant 2$.\\
	
	$$\begin{tikzpicture}[scale=0.75]
	\draw [-> , line width = 1pt] (-8,0) -- (8,0) ;
	\draw (0,0) node[below=6pt]{$0$};
	\draw (1,0) node[below=6pt]{$1$};
	\foreach \x in {-7,...,7}
	{
		%	\draw (\x,0) node[below=3pt]{$\x$};
		\draw [line width = 1pt] (\x,-0.1) -- (\x,0.1) ;
	}
	\draw (-8,0) node[left=3pt]{$-\infty$};
	\draw (8,0) node[right=3pt]{$+\infty$};
	\end{tikzpicture}$$
	
	On note cet intervalle \ldots \ldots \ldots \ldots \\[0.2cm]
	
	\begin{rmq}
		Attention, le \ldots \ldots \ldots en \ldots \ldots \ldots est toujours \ldots \ldots \ldots \ldots car \ldots \ldots \ldots \ldots \ldots \ldots \ldots \ldots \ldots \ldots \\[0.2cm]
		.\dotfill\\
	\end{rmq}
	
	\newpage
	
	
	\item Repasser en couleur sur cette droite l'ensemble des points $M$ dont l'abscisse vérifie $-4 < x < 2$.\\
	
	$$\begin{tikzpicture}[scale=0.75]
	\draw [-> , line width = 1pt] (-8,0) -- (8,0) ;
	\draw (0,0) node[below=3pt]{$0$};
	\draw (1,0) node[below=3pt]{$1$};
	\foreach \x in {-7,...,7}
	{
		%	\draw (\x,0) node[below=3pt]{$\x$};
		\draw [line width = 1pt] (\x,-0.1) -- (\x,0.1) ;
	}
	\draw (-8,0) node[left=3pt]{$-\infty$};
	\draw (8,0) node[right=3pt]{$+\infty$};
	\end{tikzpicture}$$
	
	On note cet intervalle \ldots \ldots \ldots \ldots \\
	
	\item Parmi les encadrements suivants, entourer celui qui permet de \textbf{caractériser} l'intervalle représenté :\\
	
	$$\begin{tikzpicture}[scale=0.75]
	\draw [-> , line width = 1pt] (-8,0) -- (8,0) ;
	\draw [ red, [-[ , line width = 2pt] (4,0) -- (-3.15,0) ; 
	%\draw [ red, line width = 2pt](-3,0) node{$]$};
	%\draw [ red, text width = 2pt](4,0) node{$]$};
	\foreach \x in {-7,...,7}
	{
		\draw (\x,0) node[below=3pt]{$\x$};
		\draw [line width = 1pt] (\x,-0.1) -- (\x,0.1) ;
	}
	\draw (-8,0) node[left=3pt]{$-\infty$};
	\draw (8,0) node[right=3pt]{$+\infty$};
	\end{tikzpicture}$$

	\noindent Encadrements proposés : 
	$$-3 < x < 4 \ \ \ \text{ou} \ \ \ -3 \leqslant x < 4 \ \ \ \text{ou} \ \ \ -3 < x \leqslant 4 \ \ \ \text{ou} \ \ \ -3 \leqslant x \leqslant 4$$
	


	\item Parmi les inégalités/encadrements suivants, entourer celle/celui qui permet de \textbf{caractériser} l'ensemble des points mis en évidence sur ce dessin :\\

$$\begin{tikzpicture}[scale=0.75]
\draw [-> , line width = 1pt] (-8,0) -- (8,0) ;
\draw [ red, -[ , line width = 2pt] (8,0) -- (-1.15,0) ; 
%\draw [ red, line width = 2pt](-3,0) node{$]$};
%\draw [ red, text width = 2pt](4,0) node{$]$};
\foreach \x in {-7,...,7}
{
	\draw (\x,0) node[below=3pt]{$\x$};
	\draw [line width = 1pt] (\x,-0.1) -- (\x,0.1) ;
}
\draw (-8,0) node[left=3pt]{$-\infty$};
\draw (8,0) node[right=3pt]{$+\infty$};
\end{tikzpicture}$$

\noindent Encadrements proposés : 
$$x < -1 \ \ \ \text{ou} \ \ \ x \leqslant -1 \ \ \ \text{ou} \ \ \ x>-1 \ \ \ \text{ou} \ \ \ x \geqslant -1$$

\end{enumerate}
\hfill\\
